**2. Accrue.Webhook.CachingBodyReader** (`accrue/lib/accrue/webhook/caching_body_reader.ex`):
Per D2-19 and the Plug.Parsers custom-body-reader pattern from the Plug moduledoc.

```elixir
defmodule Accrue.Webhook.CachingBodyReader do
  @moduledoc """
  Custom body reader for Plug.Parsers that tees the raw body into
  conn.assigns[:raw_body] for webhook signature verification.

  Only used inside the :accrue_webhook_raw_body pipeline (D2-20) —
  never globally.
  """

  def read_body(conn, opts) do
    case Plug.Conn.read_body(conn, opts) do
      {:ok, body, conn} ->
        conn = update_in(conn.assigns[:raw_body], &[body | (&1 || [])])
        {:ok, body, conn}

      {:more, body, conn} ->
        conn = update_in(conn.assigns[:raw_body], &[body | (&1 || [])])
        {:more, body, conn}

      {:error, reason} ->
        {:error, reason}
    end
  end
end
```

Note: Prepend for O(1) in streaming case; reverse at flatten time in the Plug (per RESEARCH Pattern 2 note).

**3. Accrue.Webhook.Signature** (`accrue/lib/accrue/webhook/signature.ex`):
Wrapper around `LatticeStripe.Webhook.construct_event!/4`. Accrue writes ZERO crypto code.

```elixir
defmodule Accrue.Webhook.Signature do
  @moduledoc """
  Webhook signature verification. Delegates entirely to
  `LatticeStripe.Webhook.construct_event!/4` (D2-22).

  Accrue does NOT implement HMAC verification — lattice_stripe handles
  timing-safe compare, replay tolerance, and multi-secret rotation.
  """

  @doc """
  Verifies the webhook signature and returns the parsed event.
  Raises `Accrue.SignatureError` on failure.

  `secrets` may be a single string or a list of strings for rotation (D2-05, T-2-05).
  """
  @spec verify!(binary(), String.t() | nil, String.t() | [String.t()], keyword()) ::
          LatticeStripe.Event.t()
  def verify!(raw_body, sig_header, secrets, opts \\ []) do
    tolerance = Keyword.get(opts, :tolerance, 300)
    LatticeStripe.Webhook.construct_event!(raw_body, sig_header, secrets, tolerance: tolerance)
  rescue
    e in LatticeStripe.Webhook.SignatureVerificationError ->
      reraise Accrue.SignatureError, [reason: Exception.message(e)], __STACKTRACE__
  end
end
```

**Accrue.Webhook.Plug** (`accrue/lib/accrue/webhook/plug.ex`):
```elixir
defmodule Accrue.Webhook.Plug do
  @behaviour Plug
  import Plug.Conn
  require Logger

  alias Accrue.Webhook.{Signature, CachingBodyReader}

  @impl true
  def init(opts), do: opts

  @impl true
  def call(conn, opts) do
    processor = Keyword.fetch!(opts, :processor)

    :telemetry.span([:accrue, :webhook, :receive], %{processor: processor}, fn ->
      result = do_call(conn, processor)
      {result, %{processor: processor}}
    end)
  rescue
    e in Accrue.SignatureError ->
      Logger.warning("Webhook signature verification failed: #{e.reason}")
      conn |> send_resp(400, Jason.encode!(%{error: "signature_verification_failed"})) |> halt()
  end

  defp do_call(conn, processor) do
    raw_body = flatten_raw_body(conn)
    sig_header = get_req_header(conn, "stripe-signature") |> List.first()

    unless sig_header do
      raise Accrue.SignatureError, reason: "missing stripe-signature header"
    end

    secrets = Accrue.Config.webhook_signing_secrets(processor)
    event = Signature.verify!(raw_body, sig_header, secrets)

    # Delegate to Accrue.Webhook.Ingest (Plan 04) for persist + enqueue.
    # For now, store event in conn.private for Plan 04 to wire.
    conn
    |> put_private(:accrue_verified_event, event)
    |> put_private(:accrue_raw_body, raw_body)
    |> put_private(:accrue_processor, processor)
    # Plan 04 will replace this with the actual ingest call.
    # Temporary: return 200 so plug tests pass.
    |> send_resp(200, Jason.encode!(%{ok: true}))
    |> halt()
  end

  defp flatten_raw_body(conn) do
    case conn.assigns[:raw_body] do
      nil -> ""
      chunks when is_list(chunks) -> chunks |> Enum.reverse() |> IO.iodata_to_binary()
      binary when is_binary(binary) -> binary
    end
  end
end
```

NOTE: The `send_resp(200, ...)` is temporary. Plan 04 will replace the final section of `do_call/2` with `Accrue.Webhook.Ingest.run(conn, processor, event, raw_body)` which does the transactional persist+enqueue+respond. The plug test suite verifies signature behavior independent of persistence.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Internet -> Webhook Plug | Untrusted HTTP POST with potentially forged signature |
| Stripe-Signature header -> Signature.verify! | Header value could be tampered or replayed |
| Raw body -> signature input | Body must be byte-exact (no re-encoding) for HMAC to match |

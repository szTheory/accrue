**Accrue.WebhookFixtures** (`accrue/test/support/webhook_fixtures.ex`):
Generates signed webhook payloads using lattice_stripe's test signature helper.

```elixir
defmodule Accrue.WebhookFixtures do
  @moduledoc """
  Test fixtures for webhook payloads. Generates valid Stripe-signed
  webhook events using lattice_stripe's test signature helper.
  """

  @default_secret "whsec_test_secret_for_accrue_tests"

  def default_secret, do: @default_secret

  @doc """
  Builds a signed webhook request body and stripe-signature header.
  Returns {body_string, signature_header}.
  """
  def signed_event(attrs \\ %{}, opts \\ []) do
    secret = Keyword.get(opts, :secret, @default_secret)
    event = build_event(attrs)
    body = Jason.encode!(event)
    signature = LatticeStripe.Webhook.generate_test_signature(body, secret)
    {body, signature}
  end

  @doc "Builds a webhook event map (unsigned)."
  def build_event(overrides \\ %{}) do
    Map.merge(
      %{
        "id" => "evt_test_#{System.unique_integer([:positive])}",
        "object" => "event",
        "type" => "customer.created",
        "livemode" => false,
        "created" => DateTime.utc_now() |> DateTime.to_unix(),
        "data" => %{
          "object" => %{
            "id" => "cus_test_#{System.unique_integer([:positive])}",
            "object" => "customer",
            "email" => "test@example.com"
          }
        },
        "api_version" => "2026-03-25.dahlia"
      },
      overrides
    )
  end

  @doc "Builds a tampered body (valid JSON but signature won't match)."
  def tampered_event(attrs \\ %{}, opts \\ []) do
    {_body, signature} = signed_event(attrs, opts)
    tampered_body = Jason.encode!(Map.put(build_event(attrs), "tampered", true))
    {tampered_body, signature}
  end
end
```

Create StreamData property tests for money arithmetic edge cases. This fulfills TEST-09 (Oban.Testing integration) in conjunction with Task 1's Oban config, and provides the property test scaffold mentioned in VALIDATION.md Wave 0.

**Property tests to implement:**

1. **Zero-decimal currency safety:** For currencies like JPY, KRW, amounts are always whole numbers. Property: for any amount in a zero-decimal currency, `Money.new(amount, currency).amount` has no fractional part.

2. **Addition commutativity:** `Money.add(a, b) == Money.add(b, a)` for same-currency money values.

3. **Addition associativity:** `Money.add(Money.add(a, b), c) == Money.add(a, Money.add(b, c))` for same-currency.

4. **Mixed-currency rejection:** `Money.add(money_usd, money_eur)` raises `Accrue.CurrencyMismatchError` or returns `{:error, :currency_mismatch}`.

5. **Roundtrip through integer cents:** `Money.new(cents, :usd) |> Money.to_cents()` equals the original cents value.

Use `StreamData.integer/0` for amounts, `StreamData.member_of/1` for currencies from a known set.

```elixir
defmodule Accrue.Property.MoneyPropertyTest do
  use ExUnit.Case, async: true
  use ExUnitProperties

  alias Accrue.Money

  property "zero-decimal currencies have integer amounts" do
    check all amount <- integer(1..999_999),
              currency <- member_of([:jpy, :krw, :vnd, :bif, :clp, :djf]) do
      money = Money.new(amount, currency)
      assert money  # verify it creates without error
    end
  end

  property "same-currency addition is commutative" do
    check all a <- integer(1..999_999),
              b <- integer(1..999_999) do
      ma = Money.new(a, :usd)
      mb = Money.new(b, :usd)
      assert Money.add(ma, mb) == Money.add(mb, ma)
    end
  end

  # Additional properties...
end
```

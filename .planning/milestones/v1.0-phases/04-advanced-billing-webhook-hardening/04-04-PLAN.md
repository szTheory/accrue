<objective>
Implement BILL-15 dunning with the D4-02 hybrid strategy: Stripe Smart Retries owns the retry cadence; Accrue owns a thin grace-period overlay. `Accrue.Billing.Dunning` is a pure policy module. `Accrue.Jobs.DunningSweeper` is an Oban cron that calls Stripe to flip past_due subscriptions to unpaid/canceled after the grace window. The `customer.subscription.updated` webhook handler emits terminal telemetry when Stripe echoes back the status change. Local status is NEVER touched by the sweeper (D2-29 canonicality).

Purpose: close the critical revenue-recovery gap. Without dunning, failed payments quietly expire without operator visibility. The hybrid model avoids Stripe dashboard lock-in while preserving Stripe as the state authority.

Output: pure policy module, Oban sweeper, webhook-driven terminal telemetry, property tests.
</objective>

  defmodule Accrue.Billing.Dunning do
    @moduledoc """
    Pure policy module for BILL-15 dunning (D4-02 hybrid).
    No side effects, no DB, no Stripe calls — Accrue.Jobs.DunningSweeper holds those.
    """
    alias Accrue.Billing.Subscription

    @type decision :: {:sweep, :unpaid | :canceled} | :hold | :skip
    @type policy :: keyword()

    @spec compute_terminal_action(Subscription.t(), policy()) :: decision()
    def compute_terminal_action(%Subscription{} = sub, policy) do
      cond do
        Keyword.get(policy, :mode) == :disabled -> :skip
        sub.status != :past_due -> :skip
        not is_nil(sub.dunning_sweep_attempted_at) -> :skip
        grace_elapsed?(sub.past_due_since, Keyword.fetch!(policy, :grace_days), DateTime.utc_now()) ->
          {:sweep, Keyword.fetch!(policy, :terminal_action)}
        true -> :hold
      end
    end

    @spec grace_elapsed?(DateTime.t() | nil, pos_integer(), DateTime.t()) :: boolean()
    def grace_elapsed?(nil, _grace_days, _now), do: false
    def grace_elapsed?(%DateTime{} = past_due_since, grace_days, now) do
      DateTime.diff(now, past_due_since, :second) > grace_days * 86_400
    end
  end
  ```

  defmodule Accrue.Jobs.DunningSweeper do
    @moduledoc """
    Oban cron worker — BILL-15 / D4-02 grace-period overlay.

    Calls Stripe to transition past_due subscriptions to unpaid/canceled once grace window elapses.
    NEVER flips local status (D2-29 — Stripe is canonical; webhook flips the row).

    Host wires cron:
        {Oban.Plugins.Cron, crontab: [{"*/15 * * * *", Accrue.Jobs.DunningSweeper}]}
    """
    use Oban.Worker, queue: :accrue_dunning, max_attempts: 3
    require Logger

    alias Accrue.{Config, Events, Processor, Repo}
    alias Accrue.Billing.{Dunning, Query, Subscription}

    @impl Oban.Worker
    def perform(%Oban.Job{} = job) do
      Accrue.Oban.Middleware.put(job)
      sweep()
    end

    def sweep do
      policy = Config.fetch!(:dunning)
      case Keyword.get(policy, :mode) do
        :disabled -> {:ok, 0}
        _ ->
          grace_days = Keyword.fetch!(policy, :grace_days)
          candidates = grace_days |> Query.dunning_sweep_candidates() |> Repo.all()
          count = Enum.reduce(candidates, 0, fn sub, acc ->
            case Dunning.compute_terminal_action(sub, policy) do
              {:sweep, terminal} -> if sweep_one(sub, terminal), do: acc + 1, else: acc
              _ -> acc
            end
          end)
          {:ok, count}
      end
    end

    defp sweep_one(%Subscription{} = sub, terminal_action) do
      stripe_status = Atom.to_string(terminal_action)
      case Processor.__impl__().subscription_update(sub.processor_id, %{status: stripe_status}, []) do
        {:ok, _stripe_sub} ->
          # Stamp attempt AFTER successful Stripe call. Does NOT flip local status.
          {:ok, _} =
            sub
            |> Ecto.Changeset.change(%{dunning_sweep_attempted_at: DateTime.utc_now()})
            |> Repo.update()
          Events.record(%{
            type: "dunning.terminal_action_requested",
            subject_type: "Subscription",
            subject_id: sub.id,
            data: %{terminal_action: terminal_action, mode: :accrue_sweeper}
          })
          true
        {:error, err} ->
          Logger.warning("DunningSweeper Stripe error for sub #{sub.id}: #{inspect(err)}")
          false
      end
    end
  end
  ```

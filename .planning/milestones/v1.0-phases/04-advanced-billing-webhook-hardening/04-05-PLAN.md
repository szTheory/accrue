  ```elixir
  defmodule Accrue.Billing.PromotionCode do
    use Ecto.Schema
    import Ecto.Changeset
    alias Accrue.Billing.Metadata

    @primary_key {:id, :binary_id, autogenerate: true}
    @foreign_key_type :binary_id

    schema "accrue_promotion_codes" do
      field :processor, :string, default: "stripe"
      field :processor_id, :string
      field :code, :string
      belongs_to :coupon, Accrue.Billing.Coupon, type: :binary_id
      field :active, :boolean, default: true
      field :max_redemptions, :integer
      field :times_redeemed, :integer, default: 0
      field :expires_at, :utc_datetime_usec
      field :data, :map, default: %{}
      field :metadata, :map, default: %{}
      field :lock_version, :integer, default: 1
      timestamps(type: :utc_datetime_usec)
    end

    @cast_fields ~w[processor processor_id code coupon_id active max_redemptions times_redeemed expires_at data metadata]a
    @required_fields ~w[processor processor_id code]a

    def changeset(promo_or_changeset, attrs \\ %{}) do
      promo_or_changeset
      |> cast(attrs, @cast_fields)
      |> validate_required(@required_fields)
      |> Metadata.validate_metadata(:metadata)
      |> optimistic_lock(:lock_version)
      |> unique_constraint(:processor_id)
      |> unique_constraint(:code)
      |> foreign_key_constraint(:coupon_id)
    end

    def force_status_changeset(promo, attrs \\ %{}) do
      promo |> cast(attrs, @cast_fields) |> optimistic_lock(:lock_version) |> unique_constraint(:processor_id) |> unique_constraint(:code)
    end
  end
  ```

  ```elixir
  defmodule Accrue.Billing.CouponActions do
    alias Accrue.Billing.{Coupon, PromotionCode, PromotionCodeProjection, Subscription}
    alias Accrue.{Events, Processor, Repo}
    alias Accrue.Processor.Idempotency

    # Coupon.create/2 ŌĆö thin wrapper
    def create_coupon(params, opts \\ []) do
      idem = Idempotency.key(:coupon_create, params[:id] || :random, Accrue.Context.operation_id())
      Repo.transact(fn ->
        with {:ok, stripe_coupon} <- Processor.__impl__().coupon_create(params, idempotency_key: idem),
             attrs <- project_coupon(stripe_coupon),
             {:ok, row} <- %Coupon{} |> Coupon.changeset(attrs) |> Repo.insert(on_conflict: :nothing, conflict_target: :processor_id),
             {:ok, _} <- Events.record(%{type: "coupon.created", subject_type: "Coupon", subject_id: row.id, data: %{}}) do
          {:ok, row}
        end
      end)
    end

    def create_promotion_code(params, opts \\ []) do
      idem = Idempotency.key(:promotion_code_create, params[:code], Accrue.Context.operation_id())
      Repo.transact(fn ->
        with {:ok, stripe_promo} <- Processor.__impl__().promotion_code_create(params, idempotency_key: idem),
             {:ok, attrs} <- PromotionCodeProjection.decompose(stripe_promo),
             {:ok, row} <- %PromotionCode{} |> PromotionCode.changeset(attrs) |> Repo.insert(),
             {:ok, _} <- Events.record(%{type: "promotion_code.created", subject_type: "PromotionCode", subject_id: row.id, data: %{code: row.code}}) do
          {:ok, row}
        end
      end)
    end

    @spec apply_promotion_code(Subscription.t(), String.t()) ::
            {:ok, Subscription.t()} | {:error, :not_found | :inactive | :expired | :max_redemptions_reached | term()}
    def apply_promotion_code(%Subscription{} = sub, code) when is_binary(code) do
      with {:ok, promo} <- fetch_applicable(code),
           {:ok, coupon} <- fetch_coupon(promo.coupon_id),
           {:ok, _stripe_sub} <-
             Processor.__impl__().subscription_update(
               sub.processor_id,
               %{coupon: coupon.processor_id},
               idempotency_key: Idempotency.key(:apply_promotion, sub.id, Accrue.Context.operation_id())
             ) do
        Events.record(%{
          type: "coupon.applied",
          subject_type: "Subscription",
          subject_id: sub.id,
          data: %{coupon_id: coupon.processor_id, promotion_code: code}
        })
        {:ok, sub}
      end
    end

    defp fetch_applicable(code) do
      now = DateTime.utc_now()
      case Repo.get_by(PromotionCode, code: code) do
        nil -> {:error, :not_found}
        %PromotionCode{active: false} -> {:error, :inactive}
        %PromotionCode{expires_at: exp} when not is_nil(exp) ->
          if DateTime.compare(exp, now) == :lt, do: {:error, :expired}, else: {:ok, _check_redemptions()}
        %PromotionCode{max_redemptions: max, times_redeemed: redeemed} = promo
          when is_integer(max) and redeemed >= max -> {:error, :max_redemptions_reached}
        promo -> {:ok, promo}
      end
    end

    defp fetch_coupon(coupon_id), do: case Repo.get(Coupon, coupon_id) do
      nil -> {:error, :coupon_missing}
      c -> {:ok, c}
    end

    defp project_coupon(stripe_coupon), do: # standard stringify + field extraction
  end
  ```

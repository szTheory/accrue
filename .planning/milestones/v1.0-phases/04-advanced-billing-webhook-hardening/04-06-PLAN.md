  defmodule Accrue.Webhooks.DLQ do
    @moduledoc """
    Dead-letter queue replay and retention for webhook events (WH-08 / D4-04).

    Replay MUST insert fresh Oban jobs via `Oban.insert/2` — `Oban.retry_job/2` refuses
    `:discarded`/`:cancelled` states and dead-lettered events are exactly those.
    """
    import Ecto.Query
    alias Accrue.{Config, Events, Repo}
    alias Accrue.Webhook.WebhookEvent

    @type filter :: [type: String.t() | [String.t()], since: DateTime.t(),
                     until: DateTime.t(), livemode: boolean()]

    @type replay_opts :: [batch_size: pos_integer(), stagger_ms: non_neg_integer(),
                          dry_run: boolean(), force: boolean()]

    @spec requeue(Ecto.UUID.t()) :: {:ok, WebhookEvent.t()} | {:error, Accrue.Error.t()}
    def requeue(id) when is_binary(id) do
      Repo.transact(fn ->
        with {:ok, row} <- fetch_replayable(id),
             {:ok, updated} <- row |> WebhookEvent.status_changeset(:received) |> Repo.update(),
             {:ok, _job} <- Oban.insert(
               Accrue.Webhook.DispatchWorker.new(%{"webhook_event_id" => updated.id})
             ),
             {:ok, _evt} <- Events.record(%{
               type: "webhook.replay_requested",
               subject_type: "WebhookEvent",
               subject_id: updated.id,
               actor_type: :admin,
               data: %{original_processor_event_id: row.processor_event_id},
               idempotency_key: "replay:" <> row.processor_event_id
             }) do
          emit_replay_telemetry(%{event_id: id, actor: :replay})
          {:ok, updated}
        end
      end)
    end

    def requeue!(id), do: unwrap!(requeue(id))

    @spec requeue_where(filter(), replay_opts()) ::
            {:ok, %{requeued: non_neg_integer(), skipped: non_neg_integer()}} | {:error, term()}
    def requeue_where(filter, opts \\ []) do
      batch_size = Keyword.get(opts, :batch_size, Config.fetch!(:dlq_replay_batch_size))
      stagger_ms = Keyword.get(opts, :stagger_ms, Config.fetch!(:dlq_replay_stagger_ms))
      dry_run? = Keyword.get(opts, :dry_run, false)
      force? = Keyword.get(opts, :force, false)
      max_rows = Config.fetch!(:dlq_replay_max_rows)

      query = build_query(filter) |> where([w], w.status in [:dead, :failed])
      total = Repo.aggregate(query, :count, :id)

      cond do
        total > max_rows and not force? ->
          {:error, :replay_too_large}
        dry_run? ->
          {:ok, %{requeued: 0, skipped: 0, would_requeue: total}}
        true ->
          bulk_requeue(query, batch_size, stagger_ms)
      end
    end

    def requeue_where!(filter, opts \\ []), do: unwrap!(requeue_where(filter, opts))

    @spec list(filter(), keyword()) :: [WebhookEvent.t()]
    def list(filter, opts \\ []) do
      limit = Keyword.get(opts, :limit, 100)
      offset = Keyword.get(opts, :offset, 0)
      filter |> build_query() |> limit(^limit) |> offset(^offset) |> Repo.all()
    end

    @spec count(filter()) :: non_neg_integer()
    def count(filter), do: filter |> build_query() |> Repo.aggregate(:count, :id)

    @spec prune(pos_integer() | :infinity) :: {:ok, non_neg_integer()}
    def prune(:infinity), do: {:ok, 0}
    def prune(days) when is_integer(days) and days > 0 do
      cutoff = DateTime.add(DateTime.utc_now(), -days * 86_400, :second)
      {count, _} = from(w in WebhookEvent, where: w.status == :dead and w.inserted_at < ^cutoff)
                   |> Repo.delete_all()
      {:ok, count}
    end

    @spec prune_succeeded(pos_integer() | :infinity) :: {:ok, non_neg_integer()}
    def prune_succeeded(:infinity), do: {:ok, 0}
    def prune_succeeded(days) when is_integer(days) and days > 0 do
      cutoff = DateTime.add(DateTime.utc_now(), -days * 86_400, :second)
      {count, _} = from(w in WebhookEvent, where: w.status == :succeeded and w.inserted_at < ^cutoff)
                   |> Repo.delete_all()
      {:ok, count}
    end

    # --- private ---
    defp fetch_replayable(id) do
      case Repo.get(WebhookEvent, id) do
        nil -> {:error, %Accrue.Error{type: :not_found, message: "webhook event #{id} not found"}}
        %WebhookEvent{status: :replayed} -> {:error, %Accrue.Error{type: :already_replayed, message: "already replayed"}}
        %WebhookEvent{status: s} = row when s in [:dead, :failed] -> {:ok, row}
        %WebhookEvent{status: s} -> {:error, %Accrue.Error{type: :not_dead_lettered, message: "status=#{s}"}}
      end
    end

    defp build_query(filter) do
      filter
      |> Enum.reduce(from(w in WebhookEvent), fn
        {:type, types}, q when is_list(types) -> where(q, [w], w.type in ^types)
        {:type, t}, q when is_binary(t) -> where(q, [w], w.type == ^t)
        {:since, ts}, q -> where(q, [w], w.inserted_at >= ^ts)
        {:until, ts}, q -> where(q, [w], w.inserted_at <= ^ts)
        {:livemode, lm}, q -> where(q, [w], w.livemode == ^lm)
      end)
    end

    defp bulk_requeue(query, batch_size, stagger_ms) do
      {requeued, skipped} =
        query
        |> Repo.all()
        |> Stream.chunk_every(batch_size)
        |> Enum.reduce({0, 0}, fn chunk, {req, skip} ->
          new_req = Enum.reduce(chunk, req, fn row, acc ->
            case requeue(row.id) do
              {:ok, _} -> acc + 1
              _ -> acc
            end
          end)
          if stagger_ms > 0, do: Process.sleep(stagger_ms)
          {new_req, skip + length(chunk) - (new_req - req)}
        end)
      {:ok, %{requeued: requeued, skipped: skipped}}
    end

    defp emit_replay_telemetry(metadata) do
      :telemetry.execute([:accrue, :ops, :webhook_dlq, :replay], %{count: 1}, metadata)
    end

    defp unwrap!({:ok, v}), do: v
    defp unwrap!({:error, err}) when is_exception(err), do: raise err
    defp unwrap!({:error, other}), do: raise "DLQ op failed: #{inspect(other)}"
  end
  ```

  defmodule Mix.Tasks.Accrue.Webhooks.Replay do
    use Mix.Task
    @shortdoc "Requeue dead-lettered webhook events"
    @moduledoc """
    Usage:
        mix accrue.webhooks.replay <event_id>
        mix accrue.webhooks.replay --since 2026-04-01 --type invoice.payment_failed [--dry-run]
        mix accrue.webhooks.replay --all-dead [--yes]
    """
    def run(argv) do
      Mix.Task.run("app.start")
      {opts, args, _} = OptionParser.parse(argv,
        strict: [since: :string, until: :string, type: :string,
                 dry_run: :boolean, all_dead: :boolean, yes: :boolean, force: :boolean])

      case args do
        [event_id] ->
          case Accrue.Webhooks.DLQ.requeue(event_id) do
            {:ok, row} -> Mix.shell().info("Requeued #{row.id}")
            {:error, err} -> Mix.raise("Requeue failed: #{inspect(err)}")
          end
        [] ->
          filter = build_filter(opts)
          replay_opts = [dry_run: opts[:dry_run] || false, force: opts[:force] || false]
          confirm_if_nuclear(opts, filter, replay_opts)
          case Accrue.Webhooks.DLQ.requeue_where(filter, replay_opts) do
            {:ok, %{requeued: n} = result} ->
              Mix.shell().info("Requeued #{n} events: #{inspect(result)}")
            {:error, :replay_too_large} ->
              Mix.raise("Replay exceeds dlq_replay_max_rows. Re-run with --force to override.")
            {:error, other} ->
              Mix.raise("Replay failed: #{inspect(other)}")
          end
      end
    end

    defp build_filter(opts) do
      []
      |> then(fn f -> if opts[:since], do: Keyword.put(f, :since, parse_dt(opts[:since])), else: f end)
      |> then(fn f -> if opts[:until], do: Keyword.put(f, :until, parse_dt(opts[:until])), else: f end)
      |> then(fn f -> if opts[:type], do: Keyword.put(f, :type, opts[:type]), else: f end)
    end

    defp parse_dt(s), do: DateTime.from_iso8601(s <> "T00:00:00Z") |> elem(1)

    defp confirm_if_nuclear(opts, filter, _replay_opts) do
      if opts[:all_dead] and not opts[:yes] do
        count = Accrue.Webhooks.DLQ.count(filter)
        if count > 10 do
          response = Mix.shell().prompt("This will requeue #{count} events. Continue? [y/N]")
          unless String.trim(response) in ["y", "Y"], do: Mix.raise("Aborted.")
        end
      end
    end
  end
  ```

  @spec timeline_for(String.t(), Ecto.UUID.t(), keyword()) :: [Event.t()]
  def timeline_for(subject_type, subject_id, opts \\ []) do
    limit = Keyword.get(opts, :limit, 1_000)
    from(e in Event,
      where: e.subject_type == ^subject_type and e.subject_id == ^subject_id,

  @spec state_as_of(String.t(), Ecto.UUID.t(), DateTime.t()) ::

        where: e.subject_type == ^subject_type and e.subject_id == ^subject_id

    {state, last_at} = Enum.reduce(events, {%{}, nil}, fn e, {acc, _} ->
      {Map.merge(acc, e.data), e.inserted_at}
    end)

    %{state: state, event_count: length(events), last_event_at: last_at}
  end

  @spec bucket_by(keyword(), atom()) :: [{DateTime.t(), non_neg_integer()}]
  def bucket_by(filter, bucket_size) when bucket_size in [:day, :week, :month] do
    bucket_str = Atom.to_string(bucket_size)
    from(e in Event,
      where: ^build_bucket_where(filter),
      group_by: fragment("date_trunc(?, ?)", ^bucket_str, e.inserted_at),
      order_by: fragment("date_trunc(?, ?)", ^bucket_str, e.inserted_at),
      select: {fragment("date_trunc(?, ?)", ^bucket_str, e.inserted_at), count(e.id)}
    )
    |> Repo.all()
  end

  defp upcast_to_current(event) do
    current = Accrue.Events.Schemas.current_version(event.type)
    case Accrue.Events.UpcasterRegistry.chain(event.type, event.schema_version, current) do
      {:ok, modules} ->
        data = Enum.reduce(modules, event.data, fn m, acc ->
          {:ok, upcasted} = m.upcast(acc)
          upcasted
        end)
        %{event | data: data, schema_version: current}
      {:error, {:unknown_schema_version, v}} ->
        :telemetry.execute([:accrue, :ops, :events_upcast_failed], %{count: 1},
          %{event_id: event.id, schema_version: v})
        raise Accrue.Error, type: :unknown_schema_version, message: "unknown schema_version #{v}"
    end
  end

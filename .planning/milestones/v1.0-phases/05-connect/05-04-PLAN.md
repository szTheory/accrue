       ```elixir
       defmodule Accrue.Connect.PlatformFee do
         @moduledoc """
         Pure Money math for platform fee computation (D5-04, CONN-06).

         Computation order (Stripe docs):
           1. Round percent component (banker's rounding at currency minor precision)
           2. Add fixed component
           3. Clamp min (floor) if present
           4. Clamp max (ceiling) if present
         """

         alias Accrue.{Config, Error, Money}

         @spec compute(Money.t(), keyword()) :: {:ok, Money.t()} | {:error, Error.t()}
         def compute(%Money{} = gross, opts \\ []) do
           with {:ok, cfg} <- resolve_config(opts),
                :ok <- validate_percent(cfg[:percent]),
                :ok <- validate_currency(gross, cfg[:fixed]),
                :ok <- validate_currency(gross, cfg[:min]),
                :ok <- validate_currency(gross, cfg[:max]) do
             cond do
               Money.zero?(gross) -> {:ok, Money.zero(gross.currency)}
               true -> do_compute(gross, cfg)
             end
           end
         end

         def compute(nil, _opts), do: {:error, %Error{message: "platform_fee/2 requires %Money{}"}}

         defp do_compute(gross, cfg) do
           percent_component = percent_of(gross, cfg[:percent])
           with_fixed = maybe_add(percent_component, cfg[:fixed])
           floored = maybe_floor(with_fixed, cfg[:min])
           ceiling = maybe_ceiling(floored, cfg[:max])
           {:ok, ceiling}
         end

         # ... private helpers using Money primitives only, no bare integer math
       end
       ```

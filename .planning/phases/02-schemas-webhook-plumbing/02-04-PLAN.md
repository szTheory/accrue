**Accrue.Webhook.Ingest** (`accrue/lib/accrue/webhook/ingest.ex`):
Orchestrates the atomic persist+enqueue. Uses `Ecto.Multi` per RESEARCH recommendation.

```elixir
defmodule Accrue.Webhook.Ingest do
  @moduledoc """
  Transactional webhook event persistence and Oban job dispatch (D2-24).

  All three writes (webhook_event row, Oban job, accrue_events ledger entry)
  succeed atomically or none do.
  """

  alias Accrue.Webhook.{WebhookEvent, DispatchWorker}
  alias Accrue.Events
  import Plug.Conn

  def run(conn, processor, stripe_event, raw_body) do
    changeset = WebhookEvent.ingest_changeset(%{
      processor: to_string(processor),
      processor_event_id: stripe_event.id,
      type: stripe_event.type,
      livemode: stripe_event.livemode,
      status: :received,
      raw_body: raw_body,
      received_at: DateTime.utc_now(),
      data: Map.from_struct(stripe_event)
    })

    multi =
      Ecto.Multi.new()
      |> Ecto.Multi.insert(:webhook_event, changeset,
           on_conflict: :nothing,
           conflict_target: [:processor, :processor_event_id],
           returning: true)
      |> Ecto.Multi.run(:dedup_check, fn _repo, %{webhook_event: row} ->
           # on_conflict: :nothing returns struct with id: nil on conflict (Pitfall 2)
           if row.id, do: {:ok, :new}, else: {:ok, :duplicate}
         end)
      |> Ecto.Multi.run(:maybe_enqueue, fn _repo, %{webhook_event: row, dedup_check: status} ->
           case status do
             :new ->
               job_changeset = DispatchWorker.new(%{webhook_event_id: row.id})
               # Use Oban.insert/2 inside Multi.run for conditional insert
               Oban.insert(job_changeset)
             :duplicate ->
               {:ok, :skipped}
           end
         end)

    # Add ledger event only for new events (not duplicates)
    multi =
      Ecto.Multi.run(multi, :maybe_ledger, fn _repo, %{webhook_event: row, dedup_check: status} ->
        case status do
          :new ->
            Events.record(%{
              type: "webhook.received",
              subject_type: "WebhookEvent",
              subject_id: to_string(row.id),
              actor_type: "webhook",
              data: %{
                processor: to_string(processor),
                event_id: stripe_event.id,
                event_type: stripe_event.type
              }
            })
          :duplicate ->
            {:ok, :skipped}
        end
      end)

    case Accrue.Repo.repo().transaction(multi) do
      {:ok, %{dedup_check: :new}} ->
        send_resp(conn, 200, Jason.encode!(%{ok: true})) |> halt()
      {:ok, %{dedup_check: :duplicate}} ->
        send_resp(conn, 200, Jason.encode!(%{ok: true})) |> halt()
      {:error, _step, reason, _changes} ->
        require Logger
        Logger.error("Webhook ingest failed: #{inspect(reason, limit: 200)}")
        send_resp(conn, 500, Jason.encode!(%{ok: false})) |> halt()
    end
  end
end
```

IMPORTANT: The `Oban.insert/2` inside `Multi.run` works because Oban 2.21 CE inserts into the same database the Multi's transaction is running on. The `maybe_enqueue` step runs within the Multi's transaction context.

ALTERNATIVE (cleaner): Use `Oban.insert/3` (the Multi-aware arity) for new events, and `Ecto.Multi.run` for the conditional skip. However, `Oban.insert/3` always inserts -- it doesn't support conditional insertion based on prior Multi step results without a `Multi.run` wrapper anyway. So the `Multi.run` + `Oban.insert/2` pattern is correct.

```elixir
defmodule Accrue.Webhook.Handler do
  @moduledoc """
  Behaviour for webhook event handlers (D2-27).

  Implement `handle_event/3` and pattern-match on the event type:

      defmodule MyApp.BillingHandler do
        use Accrue.Webhook.Handler

        def handle_event("invoice.payment_failed", event, _ctx) do
          MyApp.Slack.notify(event.object_id)
        end
      end
  """

  @callback handle_event(type :: String.t(), event :: Accrue.Webhook.Event.t(), ctx :: map()) ::
              :ok | {:error, term()}

  defmacro __using__(_opts) do
    quote do
      @behaviour Accrue.Webhook.Handler

      # D2-28: Fallthrough — unmatched events are silently ignored
      def handle_event(_type, _event, _ctx), do: :ok
      defoverridable handle_event: 3
    end
  end
end
```

```elixir
defmodule Accrue.Webhook.DispatchWorker do
  use Oban.Worker,
    queue: :accrue_webhooks,
    max_attempts: 25  # WH-05: 25 attempts before DLQ

  alias Accrue.Webhook.{WebhookEvent, Event, DefaultHandler}

  @impl Oban.Worker
  def perform(%Oban.Job{args: %{"webhook_event_id" => id}, attempt: attempt, max_attempts: max_attempts}) do
    row = Accrue.Repo.repo().get!(WebhookEvent, id)

    # Transition to :processing
    row
    |> WebhookEvent.status_changeset(%{status: :processing})
    |> Accrue.Repo.repo().update!()

    event = Event.from_webhook_event(row)
    ctx = %{attempt: attempt, max_attempts: max_attempts, webhook_event_id: id}

    # Push actor context (D2-12)
    Accrue.Actor.put_current(%{type: :webhook, id: row.processor_event_id})

    # D2-30: Default handler first (non-disableable), then user handlers
    default_result = safe_handle(DefaultHandler, event, ctx)

    user_results =
      Accrue.Config.webhook_handlers()
      |> Enum.map(fn handler -> safe_handle(handler, event, ctx) end)

    # Only re-raise if default handler failed (D2-30)
    case default_result do
      :ok ->
        mark_succeeded(row)
        :ok
      {:error, reason} ->
        mark_failed_or_dead(row, attempt, max_attempts)
        {:error, reason}
    end
  end

  defp safe_handle(handler, event, ctx) do
    handler.handle_event(event.type, event, ctx)
  rescue
    e ->
      require Logger
      Logger.error("Webhook handler #{inspect(handler)} crashed: #{Exception.format(:error, e, __STACKTRACE__)}")
      :telemetry.execute([:accrue, :webhook, :handler, :exception], %{}, %{module: handler, error: e})
      {:error, e}
  end

  defp mark_succeeded(row) do
    row
    |> WebhookEvent.status_changeset(%{status: :succeeded, processed_at: DateTime.utc_now()})
    |> Accrue.Repo.repo().update!()
  end

  defp mark_failed_or_dead(row, attempt, max_attempts) do
    status = if attempt >= max_attempts, do: :dead, else: :failed
    row
    |> WebhookEvent.status_changeset(%{status: status})
    |> Accrue.Repo.repo().update!()
  end
end
```

```elixir
defmodule Accrue.Webhook.Pruner do
  use Oban.Worker, queue: :accrue_maintenance

  alias Accrue.Webhook.WebhookEvent
  import Ecto.Query

  @impl Oban.Worker
  def perform(_job) do
    repo = Accrue.Repo.repo()
    succeeded_days = Accrue.Config.succeeded_retention_days()
    dead_days = Accrue.Config.dead_retention_days()

    unless succeeded_days == :infinity do
      cutoff = DateTime.utc_now() |> DateTime.add(-succeeded_days * 86400, :second)
      from(w in WebhookEvent, where: w.status == :succeeded and w.inserted_at < ^cutoff)
      |> repo.delete_all()
    end

    unless dead_days == :infinity do
      cutoff = DateTime.utc_now() |> DateTime.add(-dead_days * 86400, :second)
      from(w in WebhookEvent, where: w.status == :dead and w.inserted_at < ^cutoff)
      |> repo.delete_all()
    end

    :ok
  end
end
```

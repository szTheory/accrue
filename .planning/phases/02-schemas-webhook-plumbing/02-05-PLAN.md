**Extend Accrue.Processor.Stripe:**

Add private `compute_idempotency_key/3` function per D2-11:
```elixir
defp compute_idempotency_key(op, subject_id, opts) do
  seed =
    Keyword.get(opts, :operation_id) ||
      Accrue.Actor.current_operation_id() ||
      random_seed_with_warning(op, subject_id)

  raw = :crypto.hash(:sha256, "#{op}|#{subject_id}|#{seed}")
  "accr_" <> (Base.url_encode64(raw, padding: false) |> binary_part(0, 22))
end

defp random_seed_with_warning(op, subject_id) do
  seed = Ecto.UUID.generate()
  require Logger
  Logger.warning(
    "Accrue.Processor.Stripe: no operation_id seed for #{op}/#{subject_id}; " <>
    "retries will NOT be idempotent"
  )
  seed
end
```

```elixir
def update_customer(%Customer{} = customer, attrs) do
  multi =
    Ecto.Multi.new()
    |> Ecto.Multi.update(:customer, Customer.changeset(customer, attrs))
    |> Events.record_multi(:event, %{
      type: "customer.updated",
      subject_type: "Customer",
      subject_id: customer.id,
      data: %{changes: Map.take(attrs, [:metadata, :name, :email])}
    })

  Accrue.Repo.repo().transaction(multi)
end
```

```elixir
def put_data(%{__struct__: schema} = record, new_data) when is_map(new_data) do
  record
  |> Ecto.Changeset.change(data: new_data)
  |> Ecto.Changeset.optimistic_lock(:lock_version)
  |> Accrue.Repo.repo().update()
end

def patch_data(%{__struct__: schema} = record, partial_data) when is_map(partial_data) do
  merged = Map.merge(record.data || %{}, partial_data)
  put_data(record, merged)
end
```

    ```elixir
    defmodule Accrue.Credo.NoRawStatusAccess do
      @moduledoc false
      use Credo.Check,
        base_priority: :high,
        category: :warning,
        explanations: [
          check: """
          Never gate features on raw subscription.status — use Accrue.Billing.Subscription
          predicates (active?/1, canceling?/1, canceled?/1, past_due?/1, paused?/1, trialing?/1)
          or Accrue.Billing.Query fragments. Enforces BILL-05.
          """
        ]

      @stripe_statuses ~w(trialing active past_due canceled unpaid incomplete incomplete_expired paused)a

      @impl true
      def run(%SourceFile{} = source_file, params \\ []) do
        issue_meta = IssueMeta.for(source_file, params)
        module_name = Credo.Code.Module.name(source_file)

        if exempt_module?(module_name) do
          []
        else
          Credo.Code.prewalk(source_file, &traverse(&1, &2, issue_meta))
        end
      end

      defp exempt_module?(nil), do: true
      defp exempt_module?(name), do: String.contains?(name, "Accrue.Billing.Subscription")

      # {:., _, [{_, _, _}, :status]} — the .status field access
      defp traverse({:==, meta, [{{:., _, [_, :status]}, _, _}, _]} = node, issues, issue_meta) do
        {node, [issue_for(issue_meta, meta[:line] || 0, "== on .status") | issues]}
      end
      defp traverse({:in, meta, [{{:., _, [_, :status]}, _, _}, _]} = node, issues, issue_meta) do
        {node, [issue_for(issue_meta, meta[:line] || 0, "in on .status") | issues]}
      end
      defp traverse({:==, meta, [_, status]} = node, issues, issue_meta) when status in @stripe_statuses do
        {node, [issue_for(issue_meta, meta[:line] || 0, "== with Stripe status atom") | issues]}
      end
      defp traverse(ast, issues, _), do: {ast, issues}

      defp issue_for(issue_meta, line_no, trigger) do
        format_issue(issue_meta,
          message: "Raw subscription.status access (#{trigger}); use Accrue.Billing.Subscription predicates or Query fragments (BILL-05)",
          line_no: line_no
        )
      end
    end
    ```

    defmodule Accrue.Billing.Query do
      @moduledoc """
      Composable Ecto query fragments mirroring Accrue.Billing.Subscription predicates.
      Use these in `where` clauses instead of accessing raw `.status` (BILL-05, D3-04).
      """
      import Ecto.Query

      @spec active(Ecto.Queryable.t()) :: Ecto.Query.t()
      def active(query \\ Accrue.Billing.Subscription) do
        from s in query, where: s.status in [:active, :trialing]
      end

      def trialing(query \\ Accrue.Billing.Subscription) do
        from s in query, where: s.status == :trialing
      end

      def canceling(query \\ Accrue.Billing.Subscription) do
        now = Accrue.Clock.utc_now()
        from s in query,
          where: s.status == :active and s.cancel_at_period_end == true and
                 s.current_period_end > ^now
      end

      def canceled(query \\ Accrue.Billing.Subscription) do
        from s in query,
          where: s.status in [:canceled, :incomplete_expired] or not is_nil(s.ended_at)
      end

      def past_due(query \\ Accrue.Billing.Subscription) do
        from s in query, where: s.status in [:past_due, :unpaid]
      end

      def paused(query \\ Accrue.Billing.Subscription) do
        from s in query,
          where: s.status == :paused or not is_nil(s.pause_collection)
      end
    end

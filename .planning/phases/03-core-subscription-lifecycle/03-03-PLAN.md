    - `Accrue.Processor.Idempotency.key(op, subject_id, operation_id, sequence \\ 0)` returns a 64-char hex string.
    - Same `{op, subject_id, operation_id, sequence}` → same key. Different anything → different key.
    - Output format: `"#{op}_#{sha256_hex}"` where `sha256_hex = sha256("#{op}|#{subject_id}|#{operation_id}|#{sequence}") |> Base.encode16(case: :lower)`.

    ```elixir
    defmodule Accrue.Processor.Idempotency do
      @moduledoc """
      Deterministic idempotency key + subject UUID derivation (D3-60, D3-61, D3-64).
      Same (op, subject_id, operation_id, sequence) → same key → retries converge.
      """

      @spec key(atom() | String.t(), String.t(), String.t(), non_neg_integer()) :: String.t()
      def key(op, subject_id, operation_id, sequence \\ 0)
          when is_binary(subject_id) and is_binary(operation_id) and is_integer(sequence) and sequence >= 0 do
        op_str = to_string(op)
        hash =
          :sha256
          |> :crypto.hash("#{op_str}|#{subject_id}|#{operation_id}|#{sequence}")
          |> Base.encode16(case: :lower)
        "#{op_str}_#{hash}"
      end

      @spec subject_uuid(atom() | String.t(), String.t()) :: String.t()
      def subject_uuid(op, operation_id) when is_binary(operation_id) do
        op_str = to_string(op)
        raw = :crypto.hash(:sha256, "#{op_str}|#{operation_id}") |> binary_part(0, 16)
        # Force v4 UUID shape: set version nibble to 4 and variant to 10xx
        <<a::48, _ver::4, b::12, _var::2, c::62>> = raw
        bin = <<a::48, 4::4, b::12, 2::2, c::62>>
        Ecto.UUID.cast!(bin)
      end
    end
    ```

    ```elixir
    @spec advance(String.t() | nil, Keyword.t()) :: :ok
    def advance(stripe_id, opts) do
      days = Keyword.get(opts, :days, 0)
      seconds = Keyword.get(opts, :seconds, 0)
      total = days * 86400 + seconds
      new_now = DateTime.add(now(), total, :second)
      :ets.insert(@table, {:clock, new_now})

      case stripe_id && :ets.lookup(@table, {:subscription, stripe_id}) do
        [{_, sub}] ->
          trial_end_unix = sub["trial_end"]
          if trial_end_unix do
            trial_end_dt = DateTime.from_unix!(trial_end_unix)
            warn_dt = DateTime.add(trial_end_dt, -3 * 86400, :second)
            cond do
              DateTime.compare(new_now, trial_end_dt) in [:gt, :eq] ->
                updated = Map.put(sub, "status", "active")
                :ets.insert(@table, {{:subscription, stripe_id}, updated})
                if opts[:synthesize_webhooks] != false do
                  synthesize_event("customer.subscription.updated", updated)
                end
              DateTime.compare(new_now, warn_dt) in [:gt, :eq] ->
                if opts[:synthesize_webhooks] != false do
                  synthesize_event("customer.subscription.trial_will_end", sub)
                end
              true -> :ok
            end
          end
        _ -> :ok
      end
      :ok
    end
    ```

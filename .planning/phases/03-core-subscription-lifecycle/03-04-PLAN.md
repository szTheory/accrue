    ```elixir
    defmodule Accrue.Billing.IntentResult do
      @moduledoc "Wraps Stripe intent-capable responses per D3-06..08."

      @type t(ok) ::
              {:ok, ok}
              | {:ok, :requires_action, map()}
              | {:error, Accrue.Error.t()}

      @spec wrap({:ok, map()} | {:error, term()}) :: t(map())
      def wrap({:error, _} = err), do: err

      def wrap({:ok, %{"latest_invoice" => %{"payment_intent" => %{"status" => "requires_action"} = pi}} = sub}) do
        {:ok, :requires_action, pi}
        |> case do
          result -> result
        end
        _ = sub
        {:ok, :requires_action, pi}
      end

      def wrap({:ok, %{"status" => "requires_action"} = obj}) when is_map_key(obj, "client_secret") or is_map_key(obj, "next_action") do
        {:ok, :requires_action, obj}
      end

      def wrap({:ok, %{"status" => "requires_confirmation"} = obj}) do
        {:error, %Accrue.CardError{message: "requires_confirmation", stripe_error: obj}}
      end

      def wrap({:ok, %{"status" => "requires_payment_method"} = obj}) do
        {:error, %Accrue.CardError{message: "requires_payment_method", stripe_error: obj}}
      end

      def wrap({:ok, obj}), do: {:ok, obj}
    end
    ```

    ```elixir
    defmodule Accrue.Billing.Trial do
      @moduledoc "Trial end normalization per D3-38."

      @spec normalize_trial_end(term()) :: String.t() | pos_integer()
      def normalize_trial_end(:now), do: "now"
      def normalize_trial_end(%DateTime{} = dt), do: DateTime.to_unix(dt)
      def normalize_trial_end({:days, n}) when is_integer(n) and n > 0 do
        Accrue.Clock.utc_now() |> DateTime.add(n * 86400, :second) |> DateTime.to_unix()
      end
      def normalize_trial_end(%Duration{} = d) do
        Accrue.Clock.utc_now() |> DateTime.shift(d) |> DateTime.to_unix()
      end
      def normalize_trial_end(int) when is_integer(int) do
        raise ArgumentError, "unix ints rejected; use :now, %DateTime{}, {:days, N}, or %Duration{}"
      end
      def normalize_trial_end(:trial_period_days) do
        raise ArgumentError, "use {:days, N} sugar instead of :trial_period_days"
      end
      def normalize_trial_end(other) do
        raise ArgumentError, "invalid trial_end: #{inspect(other)}; expected :now | DateTime | {:days, N} | Duration"
      end
    end
    ```

    ```elixir
    defmodule Accrue.Billing.SubscriptionProjection do
      @moduledoc "Decomposes a Stripe subscription map to Ecto attrs per D3-02."

      @spec decompose(map()) :: {:ok, map()}
      def decompose(stripe_sub) when is_map(stripe_sub) do
        {:ok, %{
          stripe_id: stripe_sub["id"],
          status: parse_status(stripe_sub["status"]),
          cancel_at_period_end: stripe_sub["cancel_at_period_end"] || false,
          pause_collection: stripe_sub["pause_collection"],
          current_period_start: unix_to_dt(stripe_sub["current_period_start"]),
          current_period_end: unix_to_dt(stripe_sub["current_period_end"]),
          trial_start: unix_to_dt(stripe_sub["trial_start"]),
          trial_end: unix_to_dt(stripe_sub["trial_end"]),
          canceled_at: unix_to_dt(stripe_sub["canceled_at"]),
          ended_at: unix_to_dt(stripe_sub["ended_at"]),
          data: stripe_sub,
          metadata: stripe_sub["metadata"] || %{}
        }}
      end

      @valid_statuses ~w(trialing active past_due canceled unpaid incomplete incomplete_expired paused)a

      defp parse_status(nil), do: :incomplete
      defp parse_status(str) when is_binary(str) do
        atom = String.to_existing_atom(str)
        if atom in @valid_statuses, do: atom, else: :incomplete
      end

      defp unix_to_dt(nil), do: nil
      defp unix_to_dt(n) when is_integer(n), do: DateTime.from_unix!(n)
    end
    ```

    ```elixir
    defmodule Accrue.Billing.SubscriptionActions do
      alias Accrue.{Repo, Processor, Events, Clock}
      alias Accrue.Billing.{Subscription, SubscriptionItem, IntentResult, Trial, SubscriptionProjection}
      alias Accrue.Processor.Idempotency
      import Ecto.Query

      @swap_schema [
        proration: [type: {:in, [:create_prorations, :none, :always_invoice]}, required: true],
        proration_date: [type: :any, default: nil],
        billing_cycle_anchor: [type: {:in, [:unchanged, :now]}, default: :unchanged],
        payment_behavior: [type: {:in, [:default_incomplete, :pending_if_incomplete, :error_if_incomplete, :allow_incomplete]}, default: :default_incomplete],
        quantity: [type: :pos_integer, default: nil],
        metadata: [type: :map, default: nil],
        operation_id: [type: :string, default: nil],
        stripe_api_version: [type: :string, default: nil]
      ]

      @required_proration_msg "Accrue.Billing.swap_plan/3 requires an explicit :proration option (:create_prorations, :none, or :always_invoice). Accrue never inherits Stripe defaults — see BILL-09."

      def subscribe(billable, price_spec, opts \\ []) do
        price_spec = normalize_price_spec(price_spec)
        {:ok, customer} = Accrue.Billing.customer(billable)
        op_id = Keyword.get(opts, :operation_id) || Accrue.Actor.current_operation_id!()
        subject_uuid = Idempotency.subject_uuid(:create_subscription, op_id)
        idem_key = Idempotency.key(:create_subscription, subject_uuid, op_id)

        {item_params, trial_end} = build_subscribe_params(price_spec, opts)

        stripe_params = %{
          customer: customer.stripe_id,
          items: [item_params],
          payment_behavior: "default_incomplete",
          expand: ["latest_invoice.payment_intent"]
        }
        stripe_params = if trial_end, do: Map.put(stripe_params, :trial_end, trial_end), else: stripe_params
        stripe_params = maybe_put_default_pm(stripe_params, opts)

        :telemetry.span([:accrue, :billing, :subscription, :create], %{customer_id: customer.id}, fn ->
          result =
            Repo.transact(fn ->
              with {:ok, stripe_sub} <- Processor.create_subscription(stripe_params, [idempotency_key: idem_key] ++ opts),
                   {:ok, attrs} <- SubscriptionProjection.decompose(stripe_sub),
                   {:ok, sub} <- insert_subscription(subject_uuid, customer.id, attrs),
                   {:ok, _items} <- upsert_items(sub, stripe_sub),
                   {:ok, _} <- Events.record_subscription_created(sub) do
                {:ok, Repo.preload(sub, :subscription_items)}
              end
            end)
          {IntentResult.wrap(result), %{result: elem_tag(result)}}
        end)
      end

      def subscribe!(billable, price_spec, opts \\ []) do
        case subscribe(billable, price_spec, opts) do
          {:ok, sub} -> sub
          {:ok, :requires_action, pi} -> raise Accrue.ActionRequiredError, payment_intent: pi
          {:error, err} -> raise err
        end
      end

      def get_subscription(id, opts \\ []) do
        case Repo.get(Subscription, id) do
          nil -> {:error, :not_found}
          sub ->
            if Keyword.get(opts, :preload, true) do
              {:ok, Repo.preload(sub, :subscription_items)}
            else
              {:ok, sub}
            end
        end
      end

      def swap_plan(%Subscription{} = sub, new_price_id, opts) when is_binary(new_price_id) do
        validated_opts =
          case NimbleOptions.validate(opts, @swap_schema) do
            {:ok, v} -> v
            {:error, %NimbleOptions.ValidationError{message: msg}} ->
              if String.contains?(msg, "proration") do
                raise ArgumentError, @required_proration_msg
              else
                raise ArgumentError, msg
              end
          end

        sub = Repo.preload(sub, :subscription_items)
        if length(sub.subscription_items) > 1 do
          raise Accrue.Error.MultiItemSubscription,
            subscription_id: sub.id,
            item_count: length(sub.subscription_items),
            message: "swap_plan/3 supports single-item subs only; use Phase 4 update_items/3 for multi-item"
        end

        [existing_item | _] = sub.subscription_items
        op_id = validated_opts[:operation_id] || Accrue.Actor.current_operation_id!()
        idem_key = Idempotency.key(:swap_plan, sub.id, op_id)

        stripe_params = %{
          items: [%{id: existing_item.stripe_id, price: new_price_id}
                  |> maybe_put_quantity(validated_opts[:quantity])],
          proration_behavior: Atom.to_string(validated_opts[:proration]),
          expand: ["latest_invoice.payment_intent"]
        }

        Repo.transact(fn ->
          with {:ok, stripe_sub} <- Processor.update_subscription(sub.stripe_id, stripe_params, [idempotency_key: idem_key]),
               {:ok, attrs} <- SubscriptionProjection.decompose(stripe_sub),
               {:ok, updated} <- update_subscription_row(sub, attrs),
               {:ok, _items} <- upsert_items(updated, stripe_sub),
               {:ok, _} <- Events.record_plan_swapped(updated, %{new_price_id: new_price_id, proration: validated_opts[:proration]}) do
            {:ok, Repo.preload(updated, :subscription_items)}
          end
        end)
        |> IntentResult.wrap()
      end

      def swap_plan!(sub, price, opts) do
        case swap_plan(sub, price, opts) do
          {:ok, s} -> s
          {:ok, :requires_action, pi} -> raise Accrue.ActionRequiredError, payment_intent: pi
          {:error, err} -> raise err
        end
      end

      def preview_upcoming_invoice(%Subscription{} = sub, opts \\ []) do
        sub = Repo.preload(sub, :subscription_items)
        new_price_id = Keyword.get(opts, :new_price_id)
        proration = Keyword.get(opts, :proration, :create_prorations)

        items =
          case new_price_id do
            nil -> sub.subscription_items |> Enum.map(&%{id: &1.stripe_id, price: &1.price_id})
            pid -> [%{id: (hd(sub.subscription_items)).stripe_id, price: pid}]
          end

        stripe_params = %{
          customer: (Repo.preload(sub, :customer).customer).stripe_id,
          subscription: sub.stripe_id,
          subscription_details: %{items: items, proration_behavior: Atom.to_string(proration)}
        }

        with {:ok, preview} <- Processor.create_invoice_preview(stripe_params, opts),
             {:ok, upcoming} <- decompose_upcoming(preview) do
          {:ok, upcoming}
        end
      end

          raise Accrue.Error.MultiItemSubscription,
            subscription_id: sub.id, item_count: length(sub.subscription_items),

        op_id = Accrue.Actor.current_operation_id!()
        idem_key = Idempotency.key(:update_quantity, sub.id, op_id)

          with {:ok, stripe_sub} <- Processor.update_subscription(sub.stripe_id,
                 %{items: [%{id: item.stripe_id, quantity: new_quantity}]},

          end
        end)

      end

            subscription_id: sub.id,
            stripe_id: si["id"],
            price_id: get_in(si, ["price", "id"]),
            processor_plan_id: get_in(si, ["price", "id"]),
            processor_product_id: get_in(si, ["price", "product"]),
            quantity: si["quantity"] || 1,
            data: si,
            metadata: si["metadata"] || %{}
          }
          case Repo.get_by(SubscriptionItem, stripe_id: si["id"]) do
            nil -> %SubscriptionItem{} |> SubscriptionItem.changeset(attrs) |> Repo.insert!()
            existing -> existing |> SubscriptionItem.changeset(attrs) |> Repo.update!()
          end
        end)
        {:ok, :upserted}
      end

        alias Accrue.Billing.UpcomingInvoice
        currency = String.to_atom(preview["currency"] || "usd")
        lines =
          for line <- get_in(preview, ["lines", "data"]) || [] do
            %UpcomingInvoice.Line{
              description: line["description"],
              amount: Accrue.Money.new(line["amount"], currency),
              quantity: line["quantity"],
              period: period_tuple(line["period"]),
              proration?: line["proration"] == true,
              price_id: get_in(line, ["price", "id"])
            }
          end

          subscription_id: preview["subscription"],
          currency: currency,
          subtotal: Accrue.Money.new(preview["subtotal"] || 0, currency),
          total: Accrue.Money.new(preview["total"] || 0, currency),
          amount_due: Accrue.Money.new(preview["amount_due"] || 0, currency),
          starting_balance: Accrue.Money.new(preview["starting_balance"] || 0, currency),
          period_start: from_unix(preview["period_start"]),
          period_end: from_unix(preview["period_end"]),
          proration_date: from_unix(preview["subscription_proration_date"]),
          lines: lines,
          fetched_at: Accrue.Clock.utc_now()
        }}
      end

      defp period_tuple(_), do: nil
      defp from_unix(nil), do: nil
      defp from_unix(0), do: nil
      defp from_unix(n) when is_integer(n), do: DateTime.from_unix!(n)

      defp elem_tag({:ok, _}), do: :ok
      defp elem_tag({:error, _}), do: :error
    end
    ```

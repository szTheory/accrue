    defmodule Accrue.Billing.InvoiceProjection do
      @moduledoc """
      Deterministic decomposition of a Stripe invoice into Accrue schema attrs.
      Used by both user-path actions (Plan 05) and webhook DefaultHandler (Plan 07).
      """

      @valid_statuses ~w(draft open paid uncollectible void)a

      @spec decompose(map()) :: {:ok, %{invoice_attrs: map(), item_attrs: [map()]}}
      def decompose(stripe_inv) when is_map(stripe_inv) do
        invoice_attrs = %{
          stripe_id: stripe_inv["id"],
          status: parse_status(stripe_inv["status"]),
          subtotal_minor: stripe_inv["subtotal"],
          tax_minor: stripe_inv["tax"],
          discount_minor: stripe_inv["discount"] && stripe_inv["discount"]["amount_off"],
          total_minor: stripe_inv["total"],
          amount_due_minor: stripe_inv["amount_due"],
          amount_paid_minor: stripe_inv["amount_paid"],
          amount_remaining_minor: stripe_inv["amount_remaining"],
          currency: stripe_inv["currency"],
          number: stripe_inv["number"],
          hosted_url: stripe_inv["hosted_invoice_url"],
          pdf_url: stripe_inv["invoice_pdf"],
          period_start: unix_dt(stripe_inv["period_start"]),
          period_end: unix_dt(stripe_inv["period_end"]),
          due_date: unix_dt(stripe_inv["due_date"]),
          collection_method: stripe_inv["collection_method"],
          billing_reason: stripe_inv["billing_reason"],
          finalized_at: unix_dt(stripe_inv["finalized_at"]) || unix_dt(stripe_inv["status_transitions"] && stripe_inv["status_transitions"]["finalized_at"]),
          paid_at: unix_dt(stripe_inv["status_transitions"] && stripe_inv["status_transitions"]["paid_at"]),
          voided_at: unix_dt(stripe_inv["status_transitions"] && stripe_inv["status_transitions"]["voided_at"]),
          data: stripe_inv,
          metadata: stripe_inv["metadata"] || %{}
        }

        item_attrs =
          (get_in(stripe_inv, ["lines", "data"]) || [])
          |> Enum.map(fn line ->
            %{
              stripe_id: line["id"],
              description: line["description"],
              amount_minor: line["amount"],
              currency: line["currency"] || stripe_inv["currency"],
              quantity: line["quantity"] || 1,
              period_start: unix_dt(get_in(line, ["period", "start"])),
              period_end: unix_dt(get_in(line, ["period", "end"])),
              proration: line["proration"] == true,
              price_ref: get_in(line, ["price", "id"]),
              subscription_item_ref: line["subscription_item"],
              data: line
            }
          end)

        {:ok, %{invoice_attrs: invoice_attrs, item_attrs: item_attrs}}
      end

      defp parse_status(nil), do: :draft
      defp parse_status(str) when is_binary(str) do
        atom = String.to_existing_atom(str)
        if atom in @valid_statuses, do: atom, else: :draft
      end

      defp unix_dt(nil), do: nil
      defp unix_dt(0), do: nil
      defp unix_dt(n) when is_integer(n), do: DateTime.from_unix!(n)
      defp unix_dt(_), do: nil
    end
    ```

    defmodule Accrue.Billing.InvoiceActions do
      alias Accrue.{Repo, Processor, Events}
      alias Accrue.Billing.{Invoice, InvoiceItem, InvoiceProjection, IntentResult}
      alias Accrue.Processor.Idempotency

      def finalize_invoice(%Invoice{} = inv, opts \\ []),
        do: run_action(inv, :finalize_invoice, :"invoice.finalized", opts)
      def void_invoice(%Invoice{} = inv, opts \\ []),
        do: run_action(inv, :void_invoice, :"invoice.voided", opts)
      def mark_uncollectible(%Invoice{} = inv, opts \\ []),
        do: run_action(inv, :mark_uncollectible_invoice, :"invoice.marked_uncollectible", opts)
      def send_invoice(%Invoice{} = inv, opts \\ []),
        do: run_action(inv, :send_invoice, :"invoice.sent", opts)

      def pay_invoice(%Invoice{} = inv, opts \\ []) do
        result = run_action(inv, :pay_invoice, :"invoice.paid", opts)
        IntentResult.wrap(result)
      end

      # --- bang variants ---
      def finalize_invoice!(inv, opts \\ []), do: bang(finalize_invoice(inv, opts))
      def void_invoice!(inv, opts \\ []), do: bang(void_invoice(inv, opts))
      def mark_uncollectible!(inv, opts \\ []), do: bang(mark_uncollectible(inv, opts))
      def send_invoice!(inv, opts \\ []), do: bang(send_invoice(inv, opts))
      def pay_invoice!(inv, opts \\ []) do
        case pay_invoice(inv, opts) do
          {:ok, v} -> v
          {:ok, :requires_action, pi} -> raise Accrue.ActionRequiredError, payment_intent: pi
          {:error, err} -> raise err
        end
      end

      defp bang({:ok, v}), do: v
      defp bang({:error, err}), do: raise err

      # --- workflow shape ---
      defp run_action(%Invoice{} = inv, processor_fn, event_type, opts) do
        op_id = Keyword.get(opts, :operation_id) || Accrue.Actor.current_operation_id!()
        idem_key = Idempotency.key(processor_fn, inv.id, op_id)

        :telemetry.span([:accrue, :billing, :invoice, processor_fn], %{invoice_id: inv.id}, fn ->
          result =
            Repo.transact(fn ->
              with {:ok, stripe_inv} <- apply(Processor, processor_fn, [inv.stripe_id, [idempotency_key: idem_key] ++ opts]),
                   {:ok, %{invoice_attrs: attrs, item_attrs: items}} <- InvoiceProjection.decompose(stripe_inv),
                   {:ok, updated} <- update_invoice_row(inv, attrs),
                   {:ok, _} <- upsert_items(updated, items),
                   {:ok, _} <- Events.record(%{type: event_type, actor_type: :user,
                                                subject_type: "invoice", subject_id: updated.id,
                                                data: %{source: :api}}) do
                {:ok, Repo.preload(updated, :items)}
              end
            end)
          {result, %{result: tag(result)}}
        end)
      end

      defp update_invoice_row(inv, attrs) do
        inv
        |> Invoice.changeset(attrs)
        |> Repo.update()
      end

      defp upsert_items(invoice, item_attrs_list) do
        Enum.each(item_attrs_list, fn attrs ->
          attrs = Map.put(attrs, :invoice_id, invoice.id)
          case attrs.stripe_id && Repo.get_by(InvoiceItem, stripe_id: attrs.stripe_id) do
            nil -> %InvoiceItem{} |> InvoiceItem.changeset(attrs) |> Repo.insert!()
            existing -> existing |> InvoiceItem.changeset(attrs) |> Repo.update!()
          end
        end)
        {:ok, :upserted}
      end

      defp tag({:ok, _}), do: :ok
      defp tag(_), do: :error
    end
    ```

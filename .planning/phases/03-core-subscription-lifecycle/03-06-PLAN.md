    ```elixir
    defmodule Accrue.Billing.ChargeActions do
      alias Accrue.{Repo, Processor, Events, Money}
      alias Accrue.Billing.{Charge, Customer, IntentResult}
      alias Accrue.Processor.Idempotency

      def charge(billable_or_customer, %Money{} = amount, opts \\ []) do
        {:ok, customer} = resolve_customer(billable_or_customer)
        customer = Repo.preload(customer, :default_payment_method)
        pm_id = Keyword.get(opts, :payment_method) || default_pm_stripe_id(customer)

        if is_nil(pm_id) do
          {:error,
           %Accrue.Error.NoDefaultPaymentMethod{
             customer_id: customer.id,
             message: "charge/3 requires explicit :payment_method or customer.default_payment_method_id; use Accrue.Billing.set_default_payment_method/2 or pass opts[:payment_method]"
           }}
        else
          op_id = Keyword.get(opts, :operation_id) || Accrue.Actor.current_operation_id!()
          subject_uuid = Idempotency.subject_uuid(:create_charge, op_id)
          idem_key = Idempotency.key(:create_charge, subject_uuid, op_id)

          params = %{
            amount: amount.amount |> Decimal.to_integer(),
            currency: Atom.to_string(amount.currency),
            customer: customer.stripe_id,
            payment_method: pm_id,
            confirm: true,
            expand: ["balance_transaction", "payment_intent"]
          }
          params = if Keyword.has_key?(opts, :description), do: Map.put(params, :description, opts[:description]), else: params

          :telemetry.span([:accrue, :billing, :charge, :create], %{customer_id: customer.id}, fn ->
            result =
              Repo.transact(fn ->
                with {:ok, stripe_ch} <- Processor.create_charge(params, [idempotency_key: idem_key] ++ opts),
                     {:ok, charge_row} <- insert_charge(subject_uuid, customer.id, stripe_ch, amount),
                     {:ok, _} <- Events.record(%{type: charge_event_type(stripe_ch), actor_type: :user,
                                                  subject_type: "charge", subject_id: charge_row.id,
                                                  data: %{amount: amount.amount, currency: amount.currency}}) do
                  {:ok, charge_row}
                end
              end)
            {IntentResult.wrap(result), %{}}
          end)
        end
      end

      def charge!(billable, amount, opts \\ []) do
        case charge(billable, amount, opts) do
          {:ok, c} -> c
          {:ok, :requires_action, pi} -> raise Accrue.ActionRequiredError, payment_intent: pi
          {:error, err} -> raise err
        end
      end

      def create_payment_intent(params, opts \\ []) do
        Processor.create_payment_intent(params, opts) |> IntentResult.wrap()
      end

      def create_setup_intent(%Customer{} = customer, opts \\ []) do
        op_id = Keyword.get(opts, :operation_id) || Accrue.Actor.current_operation_id!()
        idem_key = Idempotency.key(:create_setup_intent, customer.id, op_id)
        params = %{customer: customer.stripe_id, usage: "off_session"}
        Processor.create_setup_intent(params, [idempotency_key: idem_key] ++ opts)
        |> IntentResult.wrap()
      end

      # --- helpers ---
      defp resolve_customer(%Customer{} = c), do: {:ok, c}
      defp resolve_customer(billable), do: Accrue.Billing.customer(billable)

      defp default_pm_stripe_id(%Customer{default_payment_method: %Accrue.Billing.PaymentMethod{stripe_id: id}}), do: id
      defp default_pm_stripe_id(_), do: nil

      defp insert_charge(id, customer_id, stripe_ch, amount) do
        bt = stripe_ch["balance_transaction"] || %{}
        fee_minor = bt["fee"]
        fees_settled_at = if fee_minor, do: Accrue.Clock.utc_now(), else: nil

        attrs = %{
          id: id,
          customer_id: customer_id,
          stripe_id: stripe_ch["id"],
          amount_minor: amount.amount |> Decimal.to_integer(),
          currency: Atom.to_string(amount.currency),
          status: stripe_ch["status"],
          stripe_fee_amount_minor: fee_minor,
          stripe_fee_currency: Atom.to_string(amount.currency),
          fees_settled_at: fees_settled_at,
          data: stripe_ch,
          metadata: stripe_ch["metadata"] || %{}
        }
        %Charge{}
        |> Charge.changeset(attrs)
        |> Ecto.Changeset.force_change(:id, id)
        |> Repo.insert()
      end

      defp charge_event_type(%{"status" => "succeeded"}), do: :"charge.succeeded"
      defp charge_event_type(%{"status" => "failed"}), do: :"charge.failed"
      defp charge_event_type(_), do: :"charge.created"
    end
    ```

    ```elixir
    defmodule Accrue.Billing.PaymentMethodActions do
      alias Accrue.{Repo, Processor, Events}
      alias Accrue.Billing.{PaymentMethod, Customer}
      alias Accrue.Processor.Idempotency
      import Ecto.Query

      def attach_payment_method(%Customer{} = customer, pm_stripe_id, opts \\ []) when is_binary(pm_stripe_id) do
        op_id = Keyword.get(opts, :operation_id) || Accrue.Actor.current_operation_id!()
        idem_key = Idempotency.key(:attach_payment_method, customer.id, op_id)

        Repo.transact(fn ->
          with {:ok, canonical} <- Processor.retrieve_payment_method(pm_stripe_id, []),
               fingerprint <- get_in(canonical, ["card", "fingerprint"]),
               {:ok, result} <- dedup_or_attach(customer, canonical, fingerprint, pm_stripe_id, idem_key, opts),
               deduped? <- Map.get(result, :existing?, false),
               {:ok, _} <- Events.record(%{type: :"payment_method.attached", actor_type: :user,
                                            subject_type: "payment_method", subject_id: result.id,
                                            data: %{deduped: deduped?, fingerprint: fingerprint}}) do
            {:ok, result}
          end
        end)
      end

      defp dedup_or_attach(customer, _canonical, nil, pm_stripe_id, idem_key, opts) do
        # No fingerprint: always insert fresh
        attach_and_insert(customer, pm_stripe_id, idem_key, opts)
      end
      defp dedup_or_attach(customer, canonical, fingerprint, pm_stripe_id, idem_key, opts) do
        case Repo.one(from p in PaymentMethod,
                      where: p.customer_id == ^customer.id and p.fingerprint == ^fingerprint) do
          nil ->
            try do
              attach_and_insert(customer, pm_stripe_id, idem_key, opts)
            rescue
              Ecto.ConstraintError ->
                # Race: another process inserted concurrently
                {:ok, _} = Processor.detach_payment_method(pm_stripe_id, [])
                existing = Repo.get_by!(PaymentMethod, customer_id: customer.id, fingerprint: fingerprint)
                {:ok, %{existing | existing?: true}}
            end
          existing ->
            # Duplicate found at application level: detach the new Stripe PM
            {:ok, _} = Processor.detach_payment_method(pm_stripe_id, [])
            {:ok, %{existing | existing?: true}}
        end
      end

      defp attach_and_insert(customer, pm_stripe_id, idem_key, opts) do
        with {:ok, attached} <- Processor.attach_payment_method(pm_stripe_id, %{customer: customer.stripe_id}, [idempotency_key: idem_key] ++ opts) do
          attrs = %{
            customer_id: customer.id,
            stripe_id: attached["id"],
            fingerprint: get_in(attached, ["card", "fingerprint"]),
            exp_month: get_in(attached, ["card", "exp_month"]),
            exp_year: get_in(attached, ["card", "exp_year"]),
            data: attached,
            metadata: attached["metadata"] || %{}
          }
          %PaymentMethod{}
          |> PaymentMethod.changeset(attrs)
          |> Repo.insert()
        end
      end

      def detach_payment_method(%PaymentMethod{} = pm) do
        Repo.transact(fn ->
          with {:ok, _} <- Processor.detach_payment_method(pm.stripe_id, []),
               {:ok, _} <- Repo.delete(pm),
               {:ok, _} <- Events.record(%{type: :"payment_method.detached", actor_type: :user,
                                            subject_type: "payment_method", subject_id: pm.id, data: %{}}) do
            {:ok, pm}
          end
        end)
      end

      def set_default_payment_method(%Customer{} = customer, %PaymentMethod{} = pm) do
        unless pm.customer_id == customer.id do
          raise Accrue.Error.NotAttached,
            customer_id: customer.id,
            payment_method_id: pm.id,
            message: "PaymentMethod #{pm.id} is not attached to Customer #{customer.id}. Call attach_payment_method/2 first."
        end

        op_id = Accrue.Actor.current_operation_id!()
        idem_key = Idempotency.key(:set_default_payment_method, customer.id, op_id)

        Repo.transact(fn ->
          with {:ok, _} <- Processor.set_default_payment_method(customer.stripe_id,
                             %{invoice_settings: %{default_payment_method: pm.stripe_id}},
                             [idempotency_key: idem_key]),
               {:ok, updated} <- customer |> Customer.changeset(%{default_payment_method_id: pm.id}) |> Repo.update(),
               {:ok, _} <- Events.record(%{type: :"customer.default_payment_method_changed",
                                            actor_type: :user, subject_type: "customer",
                                            subject_id: customer.id,
                                            data: %{payment_method_id: pm.id}}) do
            {:ok, updated}
          end
        end)
      end
    end
    ```

    ```elixir
    defmodule Accrue.Billing.RefundActions do
      alias Accrue.{Repo, Processor, Events, Money}
      alias Accrue.Billing.{Charge, Refund}
      alias Accrue.Processor.Idempotency

      def create_refund(%Charge{} = charge, opts \\ []) do
        amount_money = Keyword.get(opts, :amount)
        amount_minor =
          case amount_money do
            nil -> charge.amount_minor
            %Money{amount: a} -> Decimal.to_integer(a)
          end

        op_id = Keyword.get(opts, :operation_id) || Accrue.Actor.current_operation_id!()
        subject_uuid = Idempotency.subject_uuid(:create_refund, op_id)
        idem_key = Idempotency.key(:create_refund, subject_uuid, op_id)

        params = %{
          charge: charge.stripe_id,
          amount: amount_minor,
          expand: ["balance_transaction", "charge.balance_transaction"]
        }
        params = if opts[:reason], do: Map.put(params, :reason, opts[:reason]), else: params

        Repo.transact(fn ->
          with {:ok, stripe_refund} <- Processor.create_refund(params, [idempotency_key: idem_key]),
               {:ok, refund_row} <- insert_refund(subject_uuid, charge, stripe_refund),
               {:ok, _} <- Events.record(%{type: :"refund.created", actor_type: :user,
                                            subject_type: "refund", subject_id: refund_row.id,
                                            data: %{amount_minor: amount_minor, charge_id: charge.id}}) do
            {:ok, refund_row}
          end
        end)
      end

      def create_refund!(charge, opts \\ []) do
        case create_refund(charge, opts) do
          {:ok, r} -> r
          {:error, err} -> raise err
        end
      end

      defp insert_refund(id, charge, stripe_refund) do
        charge_bt = get_in(stripe_refund, ["charge", "balance_transaction"]) || %{}
        fee = charge_bt["fee"]
        fee_refunded = charge_bt["fee_refunded"]
        {stripe_fee_refunded, merchant_loss, settled_at} =
          case {fee, fee_refunded} do
            {nil, _} -> {nil, nil, nil}
            {_, nil} -> {nil, nil, nil}
            {f, fr} when is_integer(f) and is_integer(fr) ->
              {fr, f - fr, Accrue.Clock.utc_now()}
          end

        attrs = %{
          id: id,
          charge_id: charge.id,
          stripe_id: stripe_refund["id"],
          amount_minor: stripe_refund["amount"],
          currency: stripe_refund["currency"] || charge.currency,
          reason: stripe_refund["reason"],
          status: parse_refund_status(stripe_refund["status"]),
          stripe_fee_refunded_amount_minor: stripe_fee_refunded,
          merchant_loss_amount_minor: merchant_loss,
          fees_settled_at: settled_at,
          data: stripe_refund,
          metadata: stripe_refund["metadata"] || %{}
        }
        %Refund{}
        |> Refund.changeset(attrs)
        |> Ecto.Changeset.force_change(:id, id)
        |> Repo.insert()
      end

      defp parse_refund_status(nil), do: :pending
      defp parse_refund_status(s) when is_binary(s), do: String.to_existing_atom(s)
    end
    ```

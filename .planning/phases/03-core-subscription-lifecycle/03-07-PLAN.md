    - `handle(%{"type" => type, "id" => evt_id, "created" => created_unix, "data" => %{"object" => obj}})`:
      1. Derive `evt_ts = DateTime.from_unix!(created_unix, :second)`
      2. Wrap in `Repo.transact/1`
      3. Look up local row by `obj["id"]` and object type (via dispatch switch on `type`). If missing: upsert from canonical refetch (no stale check possible).
      4. Skip-stale: if `row.last_stripe_event_ts != nil AND evt_ts < row.last_stripe_event_ts` → emit `[:accrue, :webhooks, :stale_event]` with `%{type: type, id: evt_id}`, mark `accrue_webhook_events` row `data.stale = true`, return `{:ok, :stale}`. **Do NOT refetch.**
      5. Tie on equal timestamps → do NOT skip (D3-49).
      6. Otherwise: call `Processor.fetch(object_type_atom, obj["id"])` → decompose → `force_status_changeset` (webhook path) → update row with new `last_stripe_event_ts/id` → `Events.record` with dotted atom matching the event type (per D3-66 taxonomy).
      7. Return `{:ok, updated}`.
    - Dispatch table:
      * `customer.subscription.created / updated / trial_will_end / deleted / paused / resumed` → subscription path (fetch :subscription, decompose via SubscriptionProjection)
      * `invoice.created / finalized / paid / payment_failed / voided / marked_uncollectible / sent / upcoming` → invoice path
      * `charge.succeeded / failed / refunded / updated` → charge path (extract fee data from balance_transaction on refetch)
      * `charge.refund.updated / refund.created / refund.updated` → refund path (set fees_settled_at when fee_refunded populates)
      * `payment_method.attached / detached / updated / card_automatically_updated` → PM path
    - `trial_will_end` event emits `:"subscription.trial_ended"` telemetry span (not a schema mutation — just telemetry + Events.record).

    ```elixir
    alias Accrue.{Repo, Processor, Events}
    alias Accrue.Billing.{Subscription, SubscriptionProjection, Invoice, InvoiceProjection, InvoiceItem,
                          Charge, Refund, PaymentMethod, Customer}

    def handle(%{"type" => type, "id" => evt_id, "created" => created, "data" => %{"object" => obj}}) do
      evt_ts = DateTime.from_unix!(created, :second)
      dispatch(type, evt_id, evt_ts, obj)
    end

    # ---------------- subscription ----------------
    defp dispatch("customer.subscription." <> action, evt_id, evt_ts, obj) when action in ~w(created updated trial_will_end deleted paused resumed) do
      reduce_row(:subscription, obj["id"], evt_ts, evt_id, fn row ->
        with {:ok, canonical} <- Processor.fetch(:subscription, obj["id"]),
             {:ok, attrs} <- SubscriptionProjection.decompose(canonical),
             attrs <- Map.merge(attrs, %{last_stripe_event_ts: evt_ts, last_stripe_event_id: evt_id}),
             {:ok, updated} <- upsert_subscription(row, canonical, attrs),
             {:ok, _} <- Events.record(%{type: :"subscription.#{String.replace(action, "_", ".")}", actor_type: :webhook, subject_type: "subscription", subject_id: updated.id, data: %{source: :webhook, stripe_event_id: evt_id}}) do
          {:ok, updated}
        end
      end)
    end

    # ---------------- invoice ----------------
    defp dispatch("invoice." <> action, evt_id, evt_ts, obj) when action in ~w(created finalized paid payment_failed voided marked_uncollectible sent) do
      reduce_row(:invoice, obj["id"], evt_ts, evt_id, fn row ->
        with {:ok, canonical} <- Processor.fetch(:invoice, obj["id"]),
             {:ok, %{invoice_attrs: attrs, item_attrs: items}} <- InvoiceProjection.decompose(canonical),
             attrs <- Map.merge(attrs, %{last_stripe_event_ts: evt_ts, last_stripe_event_id: evt_id}),
             {:ok, updated} <- upsert_invoice(row, canonical, attrs),
             {:ok, _} <- upsert_invoice_items(updated, items),
             {:ok, _} <- Events.record(%{type: :"invoice.#{action}", actor_type: :webhook, subject_type: "invoice", subject_id: updated.id, data: %{source: :webhook, stripe_event_id: evt_id}}) do
          {:ok, updated}
        end
      end)
    end

    # ---------------- charge ----------------
    defp dispatch("charge." <> action, evt_id, evt_ts, obj) when action in ~w(succeeded failed updated refunded) do
      reduce_row(:charge, obj["id"], evt_ts, evt_id, fn row ->
        with {:ok, canonical} <- Processor.fetch(:charge, obj["id"]),
             {:ok, updated} <- upsert_charge(row, canonical, evt_ts, evt_id),
             {:ok, _} <- Events.record(%{type: :"charge.#{action}", actor_type: :webhook, subject_type: "charge", subject_id: updated.id, data: %{source: :webhook, stripe_event_id: evt_id}}) do
          {:ok, updated}
        end
      end)
    end

    defp dispatch("charge.refund.updated", evt_id, evt_ts, obj) do
      reduce_row(:refund, obj["id"], evt_ts, evt_id, fn row ->
        with {:ok, canonical} <- Processor.fetch(:refund, obj["id"]),
             {:ok, updated} <- upsert_refund(row, canonical, evt_ts, evt_id),
             event_type <- if(Refund.fees_settled?(updated), do: :"refund.fees_settled", else: :"refund.updated"),
             {:ok, _} <- Events.record(%{type: event_type, actor_type: :webhook, subject_type: "refund", subject_id: updated.id, data: %{source: :webhook, stripe_event_id: evt_id}}) do
          {:ok, updated}
        end
      end)
    end

    # ---------------- payment method ----------------
    defp dispatch("payment_method." <> action, evt_id, evt_ts, obj) when action in ~w(attached detached updated card_automatically_updated) do
      reduce_row(:payment_method, obj["id"], evt_ts, evt_id, fn row ->
        with {:ok, canonical} <- Processor.fetch(:payment_method, obj["id"]),
             {:ok, updated} <- upsert_payment_method(row, canonical, evt_ts, evt_id),
             event_type <- pm_event_type(action),
             {:ok, _} <- Events.record(%{type: event_type, actor_type: :webhook, subject_type: "payment_method", subject_id: updated.id, data: %{source: :webhook, stripe_event_id: evt_id}}) do
          {:ok, updated}
        end
      end)
    end

    defp pm_event_type("attached"), do: :"payment_method.attached"
    defp pm_event_type("detached"), do: :"payment_method.detached"
    defp pm_event_type("updated"), do: :"payment_method.updated"
    defp pm_event_type("card_automatically_updated"), do: :"payment_method.auto_updated"

    defp dispatch(_type, _evt_id, _evt_ts, _obj), do: {:ok, :ignored}

    # ---------------- skip-stale + reduce row shared ----------------
    defp reduce_row(object_type, stripe_id, evt_ts, evt_id, fun) do
      Repo.transact(fn ->
        row = load_row(object_type, stripe_id)
        case check_stale(row, evt_ts) do
          :stale ->
            :telemetry.execute([:accrue, :webhooks, :stale_event], %{}, %{object_type: object_type, stripe_id: stripe_id, event_id: evt_id})
            {:ok, :stale}
          :ok ->
            fun.(row)
        end
      end)
    end

    defp check_stale(nil, _evt_ts), do: :ok  # no row yet — upsert path
    defp check_stale(%{last_stripe_event_ts: nil}, _evt_ts), do: :ok
    defp check_stale(%{last_stripe_event_ts: last}, evt_ts) do
      case DateTime.compare(evt_ts, last) do
        :lt -> :stale  # strictly older → skip
        _ -> :ok       # :gt or :eq proceed (D3-49)
      end
    end

    defp load_row(:subscription, stripe_id), do: Repo.get_by(Subscription, stripe_id: stripe_id)
    defp load_row(:invoice, stripe_id), do: Repo.get_by(Invoice, stripe_id: stripe_id)
    defp load_row(:charge, stripe_id), do: Repo.get_by(Charge, stripe_id: stripe_id)
    defp load_row(:refund, stripe_id), do: Repo.get_by(Refund, stripe_id: stripe_id)
    defp load_row(:payment_method, stripe_id), do: Repo.get_by(PaymentMethod, stripe_id: stripe_id)

    defp upsert_subscription(nil, canonical, attrs) do
      # Row doesn't exist — create via webhook path (customer must exist)
      customer = Repo.get_by!(Customer, stripe_id: canonical["customer"])
      %Subscription{customer_id: customer.id}
      |> Subscription.changeset(attrs)
      |> Repo.insert()
    end
    defp upsert_subscription(row, _canonical, attrs) do
      row |> Subscription.changeset(attrs) |> Repo.update()
    end

    defp upsert_invoice(nil, canonical, attrs) do
      customer = Repo.get_by!(Customer, stripe_id: canonical["customer"])
      %Invoice{customer_id: customer.id}
      |> Invoice.force_status_changeset(attrs)
      |> Repo.insert()
    end
    defp upsert_invoice(row, _canonical, attrs) do
      row |> Invoice.force_status_changeset(attrs) |> Repo.update()
    end

    defp upsert_invoice_items(invoice, item_attrs_list) do
      Enum.each(item_attrs_list, fn attrs ->
        attrs = Map.put(attrs, :invoice_id, invoice.id)
        case attrs.stripe_id && Repo.get_by(InvoiceItem, stripe_id: attrs.stripe_id) do
          nil -> %InvoiceItem{} |> InvoiceItem.changeset(attrs) |> Repo.insert!()
          existing -> existing |> InvoiceItem.changeset(attrs) |> Repo.update!()
        end
      end)
      {:ok, :ok}
    end

    defp upsert_charge(row, canonical, evt_ts, evt_id) do
      bt = canonical["balance_transaction"] || %{}
      attrs = %{
        stripe_fee_amount_minor: bt["fee"],
        stripe_fee_currency: bt["currency"],
        fees_settled_at: if(bt["fee"], do: Accrue.Clock.utc_now(), else: nil),
        data: canonical,
        status: canonical["status"],
        last_stripe_event_ts: evt_ts,
        last_stripe_event_id: evt_id
      }
      case row do
        nil ->
          customer = Repo.get_by!(Customer, stripe_id: canonical["customer"])
          %Charge{customer_id: customer.id, stripe_id: canonical["id"],
                  amount_minor: canonical["amount"], currency: canonical["currency"], data: canonical, metadata: %{}}
          |> Charge.changeset(attrs)
          |> Repo.insert()
        existing -> existing |> Charge.changeset(attrs) |> Repo.update()
      end
    end

    defp upsert_refund(row, canonical, evt_ts, evt_id) do
      charge_bt = get_in(canonical, ["charge", "balance_transaction"]) || %{}
      fee = charge_bt["fee"]
      fee_refunded = charge_bt["fee_refunded"]
      {stripe_fee_refunded, merchant_loss, settled_at} =
        case {fee, fee_refunded} do
          {f, fr} when is_integer(f) and is_integer(fr) -> {fr, f - fr, Accrue.Clock.utc_now()}
          _ -> {nil, nil, nil}
        end
      attrs = %{
        stripe_fee_refunded_amount_minor: stripe_fee_refunded,
        merchant_loss_amount_minor: merchant_loss,
        fees_settled_at: settled_at,
        data: canonical,
        status: String.to_existing_atom(canonical["status"] || "pending"),
        last_stripe_event_ts: evt_ts,
        last_stripe_event_id: evt_id
      }
      case row do
        nil ->
          charge = Repo.get_by!(Charge, stripe_id: canonical["charge"]["id"] || canonical["charge"])
          %Refund{charge_id: charge.id, stripe_id: canonical["id"],
                  amount_minor: canonical["amount"], currency: canonical["currency"], data: canonical, metadata: %{}}
          |> Refund.changeset(attrs)
          |> Repo.insert()
        existing -> existing |> Refund.changeset(attrs) |> Repo.update()
      end
    end

    defp upsert_payment_method(row, canonical, evt_ts, evt_id) do
      attrs = %{
        fingerprint: get_in(canonical, ["card", "fingerprint"]),
        exp_month: get_in(canonical, ["card", "exp_month"]),
        exp_year: get_in(canonical, ["card", "exp_year"]),
        data: canonical,
        last_stripe_event_ts: evt_ts,
        last_stripe_event_id: evt_id
      }
      case row do
        nil ->
          customer = canonical["customer"] && Repo.get_by(Customer, stripe_id: canonical["customer"])
          %PaymentMethod{customer_id: customer && customer.id, stripe_id: canonical["id"], data: canonical, metadata: %{}}
          |> PaymentMethod.changeset(attrs)
          |> Repo.insert()
        existing -> existing |> PaymentMethod.changeset(attrs) |> Repo.update()
      end
    end
    ```

    ```elixir
      @moduledoc "Daily backstop for refund fees that weren't populated at create time (D3-46)."
      use Oban.Worker, queue: :accrue_reconcilers, max_attempts: 3
      import Ecto.Query
      alias Accrue.{Repo, Processor, Events}
      alias Accrue.Billing.Refund

      @impl Oban.Worker
      def perform(_job) do
        cutoff = DateTime.add(Accrue.Clock.utc_now(), -86400, :second)
        rows =
          from(r in Refund,
            where: is_nil(r.fees_settled_at) and r.inserted_at < ^cutoff)
          |> Repo.all()

        Enum.each(rows, &reconcile/1)
        :ok
      end

      defp reconcile(%Refund{stripe_id: sid} = row) when is_binary(sid) do
        with {:ok, canonical} <- Processor.retrieve_refund(sid, expand: ["balance_transaction", "charge.balance_transaction"]),
             charge_bt <- get_in(canonical, ["charge", "balance_transaction"]) || %{},

            stripe_fee_refunded_amount_minor: fr,
            merchant_loss_amount_minor: fee - fr,
            fees_settled_at: Accrue.Clock.utc_now(),
            data: canonical

          :telemetry.execute([:accrue, :billing, :refund, :fees_settled], %{}, %{refund_id: updated.id, source: :reconciler})

        end
      end
    ```

    ```elixir
    defmodule Accrue.Jobs.DetectExpiringCards do
      @moduledoc "Scheduled scan of payment methods for upcoming expiry (D3-71..78)."
      use Oban.Worker, queue: :accrue_scheduled, max_attempts: 3
      import Ecto.Query
      alias Accrue.{Repo, Events, Clock}
      alias Accrue.Billing.{PaymentMethod, Customer}

      @default_thresholds [30, 7, 1]

      @impl Oban.Worker
      def perform(_job) do
        thresholds = Accrue.Config.get(:expiring_card_thresholds) || @default_thresholds
        now = Clock.utc_now()

        pms =
          from(p in PaymentMethod,
            where: not is_nil(p.exp_month) and not is_nil(p.exp_year))
          |> Repo.all()

        for pm <- pms, threshold <- thresholds do
          maybe_emit(pm, threshold, now)
        end
        :ok
      end

      defp maybe_emit(pm, threshold, now) do
        expiry_dt = end_of_month(pm.exp_year, pm.exp_month)
        days_until = DateTime.diff(expiry_dt, now, :second) |> div(86400)

        if days_until in crossing_window(threshold) and not already_warned?(pm.id, threshold) do
          is_default_pm = Repo.exists?(from c in Customer, where: c.default_payment_method_id == ^pm.id)
          Events.record(%{
            type: :"card.expiring_soon", actor_type: :oban,
            subject_type: "payment_method", subject_id: pm.id,
            data: %{threshold: threshold, days_until_expiry: days_until, is_default_pm: is_default_pm}
          })
          :telemetry.execute([:accrue, :billing, :payment_method, :expiring_soon],
                              %{days_until: days_until},
                              %{payment_method_id: pm.id, threshold: threshold, is_default_pm: is_default_pm})
        end
      end

      defp crossing_window(threshold), do: [threshold]  # exact-day match; simpler than crossing logic

      defp already_warned?(pm_id, threshold) do
        one_year_ago = DateTime.add(Accrue.Clock.utc_now(), -365 * 86400, :second)
        from(e in "accrue_events",
          where: e.subject_id == ^pm_id and e.type == "card.expiring_soon"
                 and fragment("(?->>'threshold')::int = ?", e.data, ^threshold)
                 and e.inserted_at > ^one_year_ago)
        |> Repo.exists?()
      end

      defp end_of_month(year, month) do
        days = :calendar.last_day_of_the_month(year, month)
        {:ok, dt} = DateTime.new(Date.new!(year, month, days), Time.new!(23, 59, 59))
        dt
      end
    end
    ```

    ```elixir
    defmodule Accrue.Test.Factory do
      @moduledoc """
      Phase 3 test factories (D3-79..85). All route through Accrue.Processor.Fake
      and derive timestamps from Accrue.Clock.utc_now/0. Also safe to call from
      :dev via mix accrue.seed (Phase 8).
      """
      alias Accrue.{Billing, Clock, Repo}
      alias Accrue.Billing.{Customer, Subscription, SubscriptionItem}
      alias Accrue.Processor.Fake

      def customer(attrs \\ %{}) do
        owner_id = Map.get(attrs, :owner_id, Ecto.UUID.generate())
        {:ok, customer} = Billing.create_customer(Map.merge(%{
          owner_type: "User",
          owner_id: owner_id,
          email: "factory-#{owner_id}@example.com"
        }, attrs))
        %{customer: customer, owner_id: owner_id}
      end

      def subscription(attrs \\ %{}) do
        %{customer: c} = customer(attrs)
        status = Map.get(attrs, :status, :trialing)
        price = Map.get(attrs, :price_id, "price_factory_basic")
        {:ok, sub} = Billing.subscribe(c, price, trial_end: {:days, 14})
        sub =
          case status do
            :trialing -> sub
            other ->
              {:ok, _} = Fake.transition(sub.stripe_id, other, synthesize_webhooks: false)
              reload_from_canonical(sub)
          end
        %{customer: c, subscription: sub, items: sub.subscription_items}
      end

      def trialing_subscription(attrs \\ %{}), do: subscription(Map.put(attrs, :status, :trialing))
      def active_subscription(attrs \\ %{}), do: subscription(Map.put(attrs, :status, :active))
      def past_due_subscription(attrs \\ %{}), do: subscription(Map.put(attrs, :status, :past_due))
      def incomplete_subscription(attrs \\ %{}), do: subscription(Map.put(attrs, :status, :incomplete))

      def canceled_subscription(attrs \\ %{}) do
        %{customer: c, subscription: sub} = active_subscription(attrs)
        {:ok, canceled} = Billing.cancel(sub)
        %{customer: c, subscription: canceled, items: canceled.subscription_items}
      end

      def canceling_subscription(attrs \\ %{}) do
        %{customer: c, subscription: sub} = active_subscription(attrs)
        {:ok, canceling} = Billing.cancel_at_period_end(sub)
        # Ensure current_period_end is in the future (14d) for canceling?/1 predicate
        {:ok, canceling} =
          canceling
          |> Subscription.changeset(%{current_period_end: DateTime.add(Clock.utc_now(), 14 * 86400, :second)})
          |> Repo.update()
        %{customer: c, subscription: canceling, items: canceling.subscription_items}
      end

      def grace_period_subscription(attrs \\ %{}) do
        %{customer: c, subscription: sub} = canceled_subscription(attrs)
        # canceled but current_period_end in future → grace
        {:ok, grace} =
          sub
          |> Subscription.changeset(%{current_period_end: DateTime.add(Clock.utc_now(), 7 * 86400, :second)})
          |> Repo.update()
        %{customer: c, subscription: grace, items: grace.subscription_items}
      end

      def trial_ending_subscription(attrs \\ %{}) do
        %{customer: c, subscription: sub} = trialing_subscription(attrs)
        ending_dt = DateTime.add(Clock.utc_now(), 2 * 86400, :second)
        {:ok, ending} = sub |> Subscription.changeset(%{trial_end: ending_dt}) |> Repo.update()
        %{customer: c, subscription: ending, items: ending.subscription_items}
      end

      defp reload_from_canonical(sub) do
        {:ok, stripe_sub} = Fake.retrieve_subscription(sub.stripe_id, [])
        {:ok, attrs} = Accrue.Billing.SubscriptionProjection.decompose(stripe_sub)
        {:ok, updated} = sub |> Subscription.changeset(attrs) |> Repo.update()
        Repo.preload(updated, :subscription_items)
      end
    end
    ```

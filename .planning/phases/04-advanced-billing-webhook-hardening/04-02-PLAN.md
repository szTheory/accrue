<objective>
Implement BILL-13 (metered billing) end-to-end as a synchronous pass-through with a transactional-outbox audit table and a reconciler Oban worker. Closes the requirement "A developer can create a metered subscription, report meter events via `Accrue.Billing.report_usage/3`, and see usage aggregate correctly on the next invoice."

Purpose: metered billing is the cornerstone of usage-based SaaS pricing. The outbox pattern (commit row → exit txn → call Stripe → update row) is the minimum-viable durability guarantee for a billing library — without it, a crash between DB commit and HTTP call silently loses revenue events. D4-03 explicitly forbids Stripe-call-inside-transact and explicitly requires the reconciler to close the crash gap.

Output: working `Accrue.Billing.report_usage/3` + reconciler + Fake adapter + Stripe adapter + webhook handler for `billing.meter.error_report_triggered`.
</objective>

  defmodule Accrue.Billing.MeterEvent do
    use Ecto.Schema
    import Ecto.Changeset

    @primary_key {:id, :binary_id, autogenerate: true}
    @foreign_key_type :binary_id

    schema "accrue_meter_events" do
      belongs_to :customer, Accrue.Billing.Customer, type: :binary_id
      field :stripe_customer_id, :string
      field :event_name, :string
      field :value, :integer
      field :identifier, :string
      field :occurred_at, :utc_datetime_usec
      field :reported_at, :utc_datetime_usec
      field :stripe_status, :string, default: "pending"
      field :stripe_error, :map
      field :operation_id, :string
      timestamps(type: :utc_datetime_usec)
    end

    @pending_cast ~w[customer_id stripe_customer_id event_name value identifier occurred_at operation_id]a
    @pending_required ~w[stripe_customer_id event_name value identifier occurred_at]a

    def pending_changeset(attrs) do
      %__MODULE__{}
      |> cast(attrs, @pending_cast)
      |> validate_required(@pending_required)
      |> validate_number(:value, greater_than_or_equal_to: 0)
      |> put_change(:stripe_status, "pending")
      |> unique_constraint(:identifier)
    end

    def reported_changeset(row, stripe_event) do
      row
      |> change(%{
        stripe_status: "reported",
        reported_at: DateTime.utc_now(),
        stripe_error: nil
      })
    end

    def failed_changeset(row, err) do
      row
      |> change(%{
        stripe_status: "failed",
        stripe_error: normalize_error(err)
      })
    end

    defp normalize_error(%{__struct__: _} = s), do: s |> Map.from_struct() |> normalize_error()
    defp normalize_error(m) when is_map(m), do: for({k, v} <- m, into: %{}, do: {to_string(k), inspect_v(v)})
    defp normalize_error(other), do: %{"raw" => inspect(other)}
    defp inspect_v(v) when is_binary(v) or is_number(v) or is_atom(v), do: v
    defp inspect_v(v), do: inspect(v)
  end
  ```

  @spec report_usage(Accrue.Billing.Customer.t() | String.t(), String.t(), keyword()) ::
          {:ok, Accrue.Billing.MeterEvent.t()} | {:error, term()}
  def report_usage(customer, event_name, opts \\ [])

  def report_usage(stripe_customer_id, event_name, opts) when is_binary(stripe_customer_id) do
    case Repo.get_by(Accrue.Billing.Customer, processor_id: stripe_customer_id) do
      nil -> {:error, %Accrue.Error{type: :not_found, message: "customer #{stripe_customer_id} not found"}}
      customer -> report_usage(customer, event_name, opts)
    end
  end

  def report_usage(%Accrue.Billing.Customer{} = customer, event_name, opts) when is_binary(event_name) do
    opts = NimbleOptions.validate!(opts, @report_usage_schema)
    value = opts[:value]
    ts = opts[:timestamp] || DateTime.utc_now()
    identifier = opts[:identifier] || derive_identifier(customer, event_name, value, ts)

    with :ok <- validate_backdating_window(ts),
         {:ok, row} <- insert_pending(customer, event_name, value, ts, identifier, opts[:operation_id]) do
      # Stripe call OUTSIDE Repo.transact/2 — D2-09 / D4-03
      case Processor.__impl__().report_meter_event(row) do
        {:ok, stripe_event} ->
          row |> Accrue.Billing.MeterEvent.reported_changeset(stripe_event) |> Repo.update()
        {:error, err} ->
          _ = row |> Accrue.Billing.MeterEvent.failed_changeset(err) |> Repo.update()
          emit_failure_telemetry(row, err)
          {:error, err}
      end
    end
  end

  def report_usage!(customer, event_name, opts \\ []) do
    case report_usage(customer, event_name, opts) do
      {:ok, row} -> row
      {:error, err} when is_exception(err) -> raise err
      {:error, other} -> raise "report_usage!/3 failed: #{inspect(other)}"
    end
  end

  defp validate_backdating_window(ts) do
    case DateTime.diff(DateTime.utc_now(), ts, :second) do
      diff when diff > 35 * 86_400 -> {:error, :timestamp_out_of_window}
      diff when diff < -300 -> {:error, :timestamp_in_future}
      _ -> :ok
    end
  end

  defp derive_identifier(customer, event_name, value, ts) do
    op = Accrue.Context.operation_id() || "unscoped"
    hash = :erlang.phash2({customer.processor_id, event_name, value, DateTime.to_unix(ts, :microsecond)})
    "accrue_mev_#{op}_#{event_name}_#{hash}"
  end

  defp insert_pending(customer, event_name, value, ts, identifier, override_op) do
    Repo.transact(fn ->
      attrs = %{
        customer_id: customer.id,
        stripe_customer_id: customer.processor_id,
        event_name: event_name,
        value: value,
        identifier: identifier,
        occurred_at: ts,
        operation_id: override_op || Accrue.Context.operation_id()
      }
      cs = Accrue.Billing.MeterEvent.pending_changeset(attrs)

      case Repo.insert(cs) do
        {:ok, row} ->
          {:ok, _} = Accrue.Events.record_multi(:meter_event_reported, %{
            subject_type: "meter_event",
            subject_id: row.id,
            data: %{event_name: event_name, value: value, identifier: identifier}
          })
          {:ok, row}
        {:error, %Ecto.Changeset{errors: [{:identifier, _}]}} ->
          # Unique-constraint rescue: return the existing row (replay-safe)
          case Repo.get_by(Accrue.Billing.MeterEvent, identifier: identifier) do
            nil -> {:error, :identifier_race}
            existing -> {:ok, existing}
          end
        {:error, cs} ->
          {:error, cs}
      end
    end)
  end

  defp emit_failure_telemetry(row, err) do
    :telemetry.execute(
      [:accrue, :ops, :meter_reporting_failed],
      %{count: 1},
      %{meter_event_id: row.id, event_name: row.event_name, error: inspect(err)}
    )
  end
  ```

    Oban cron worker that retries meter events stuck in `pending` state more than 60 seconds.
